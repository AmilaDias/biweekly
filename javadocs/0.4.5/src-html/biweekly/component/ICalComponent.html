047 * @author Michael Angstadt
048 */
049public abstract class ICalComponent {
050        protected final ListMultimap<Class<? extends ICalComponent>, ICalComponent> components;
051        protected final ListMultimap<Class<? extends ICalProperty>, ICalProperty> properties;
052
053        public ICalComponent() {
054                components = new ListMultimap<Class<? extends ICalComponent>, ICalComponent>();
055                properties = new ListMultimap<Class<? extends ICalProperty>, ICalProperty>();
056        }

058        /**
059         * Copy constructor. Performs a deep copy of the given component's
060         * properties and sub-components.
061         * @param original the component to make a copy of
062         */
063        protected ICalComponent(ICalComponent original) {
064                properties = new ListMultimap<Class<? extends ICalProperty>, ICalProperty>(original.properties.size());
065                for (ICalProperty property : original.properties.values()) {
066                        addProperty(property.copy());
067                }
068
069                components = new ListMultimap<Class<? extends ICalComponent>, ICalComponent>(original.components.size());
070                for (ICalComponent component : original.components.values()) {
071                        addComponent(component.copy());
072                }
073        }

404        /**
405         * Checks the component for data consistency problems or deviations from the
406         * spec. These problems will not prevent the component from being written to
407         * a data stream, but may prevent it from being parsed correctly by the
408         * consuming application. These problems can largely be avoided by reading
409         * the Javadocs of the component class, or by being familiar with the
410         * iCalendar standard.
411         * @param hierarchy the hierarchy of components that the component belongs
412         * to
413         * @param version the version to validate against
414         * @see ICalendar#validate
415         * @return a list of warnings or an empty list if no problems were found
416         */
417        public final List<WarningsGroup> validate(List<ICalComponent> hierarchy, ICalVersion version) {
418                List<WarningsGroup> warnings = new ArrayList<WarningsGroup>();
419
420                //validate this component
421                List<Warning> warningsBuf = new ArrayList<Warning>(0);
422                validate(hierarchy, version, warningsBuf);
423                if (!warningsBuf.isEmpty()) {
424                        warnings.add(new WarningsGroup(this, hierarchy, warningsBuf));
425                }
426
427                //add this component to the hierarchy list
428                //copy the list so other validate() calls aren't effected
429                hierarchy = new ArrayList<ICalComponent>(hierarchy);
430                hierarchy.add(this);
431
432                //validate properties
433                for (ICalProperty property : properties.values()) {
434                        List<Warning> propWarnings = property.validate(hierarchy, version);
435                        if (!propWarnings.isEmpty()) {
436                                warnings.add(new WarningsGroup(property, hierarchy, propWarnings));
437                        }
438                }
439
440                //validate sub-components
441                for (ICalComponent component : components.values()) {
442                        warnings.addAll(component.validate(hierarchy, version));
443                }
444
445                return warnings;
446        }

466        /**
467         * Utility method for validating that there is exactly one instance of each
468         * of the given properties.
469         * @param warnings the list to add the warnings to
470         * @param classes the properties to check
471         */
472        protected void checkRequiredCardinality(List<Warning> warnings, Class<? extends ICalProperty>... classes) {
473                for (Class<? extends ICalProperty> clazz : classes) {
474                        List<? extends ICalProperty> props = getProperties(clazz);
475
476                        if (props.isEmpty()) {
477                                warnings.add(Warning.validate(2, clazz.getSimpleName()));
478                                continue;
479                        }
480
481                        if (props.size() > 1) {
482                                warnings.add(Warning.validate(3, clazz.getSimpleName()));
483                                continue;
484                        }
485                }
486        }

490         * each of the given properties.
491         * @param warnings the list to add the warnings to
492         * @param classes the properties to check

494        protected void checkOptionalCardinality(List<Warning> warnings, Class<? extends ICalProperty>... classes) {
495                for (Class<? extends ICalProperty> clazz : classes) {
496                        List<? extends ICalProperty> props = getProperties(clazz);
497
498                        if (props.size() > 1) {
499                                warnings.add(Warning.validate(3, clazz.getSimpleName()));
500                                continue;
501                        }
502                }

505        /**
506         * Utility method for validating the {@link Status} property of a component.
507         * @param warnings the list to add the warnings to
508         * @param allowed the valid statuses
509         */
510        protected void checkStatus(List<Warning> warnings, Status... allowed) {
511                Status actual = getProperty(Status.class);
512                if (actual == null) {
513                        return;
514                }
515
516                List<String> allowedValues = new ArrayList<String>(allowed.length);
517                for (Status status : allowed) {
518                        String value = status.getValue().toLowerCase();
519                        allowedValues.add(value);
520                }
521
522                String actualValue = actual.getValue().toLowerCase();
523                if (!allowedValues.contains(actualValue)) {
524                        warnings.add(Warning.validate(13, actual.getValue(), allowedValues));
525                }
526        }

528        @Override
529        public String toString() {
530                StringBuilder sb = new StringBuilder();
531                toString(0, sb);
532                return sb.toString();
533        }
534
535        private void toString(int depth, StringBuilder sb) {
536                StringUtils.repeat(' ', depth * 2, sb);
537                sb.append(getClass().getName()).append(StringUtils.NEWLINE);
538
539                depth++;
540                for (ICalProperty property : properties.values()) {
541                        StringUtils.repeat(' ', depth * 2, sb);
542                        sb.append(property).append(StringUtils.NEWLINE);
543                }
544                for (ICalComponent component : components.values()) {
545                        component.toString(depth, sb);
546                }
547        }

549        /**
550         * <p>
551         * Creates a deep copy of this component object.
552         * </p>
553         * <p>
554         * The default implementation of this method uses reflection to look for a
555         * copy constructor. Child classes SHOULD override this method to avoid the
556         * performance overhead involved in using reflection.
557         * </p>
558         * <p>
559         * The child class's copy constructor, if present, MUST invoke the
560         * {@link #ICalComponent(ICalComponent)} super constructor to ensure that
561         * the component's properties and sub-components are copied.
562         * </p>
563         * <p>
564         * This method MUST be overridden by the child class if the child class does
565         * not have a copy constructor. Otherwise, an
566         * {@link UnsupportedOperationException} will be thrown when an attempt is
567         * made to copy the component (such as in the
568         * {@link ICalendar#ICalendar(ICalendar) ICalendar class's copy constructor}
569         * ).
570         * </p>
571         * @return the copy
572         * @throws UnsupportedOperationException if the class does not have a copy
573         * constructor or there is a problem invoking it
574         */
575        public ICalComponent copy() {
576                Class<? extends ICalComponent> clazz = getClass();
577
578                try {
579                        Constructor<? extends ICalComponent> copyConstructor = clazz.getConstructor(clazz);
580                        return copyConstructor.newInstance(this);
581                } catch (Exception e) {
582                        throw new UnsupportedOperationException(Messages.INSTANCE.getExceptionMessage(1, clazz.getName()), e);
583                }
584        }

586        /**
587         * Casts all objects in the given list to the given class, adding the casted
588         * objects to a new list.
589         * @param list the list to cast
590         * @param castTo the class to cast to
591         * @return the new list
592         */
593        private static <T> List<T> castList(List<?> list, Class<T> castTo) {
594                List<T> casted = new ArrayList<T>(list.size());
595                for (Object object : list) {
596                        casted.add(castTo.cast(object));
597                }
598                return casted;
599        }

601        @Override
602        public int hashCode() {
603                final int prime = 31;
604                int result = 1;
605
606                int propertiesHash = 1;
607                for (ICalProperty property : properties.values()) {
608                        propertiesHash += property.hashCode();
609                }
610                result = prime * result + propertiesHash;
611
612                int componentsHash = 1;
613                for (ICalComponent component : components.values()) {
614                        componentsHash += component.hashCode();
615                }
616                result = prime * result + componentsHash;
617
618                return result;
619        }

621        @Override
622        public boolean equals(Object obj) {
623                if (this == obj) return true;
624                if (obj == null) return false;
625                if (getClass() != obj.getClass()) return false;
626                ICalComponent other = (ICalComponent) obj;
627
628                if (properties.size() != other.properties.size()) return false;
629                if (components.size() != other.components.size()) return false;
630
631                if (!compareMultimaps(properties, other.properties)) return false;
632                if (!compareMultimaps(components, other.components)) return false;
633
634                return true;
635        }
636
637        private static <K, V> boolean compareMultimaps(ListMultimap<K, V> map1, ListMultimap<K, V> map2) {
638                for (Map.Entry<K, List<V>> entry : map1) {
639                        K key = entry.getKey();
640                        List<V> value = entry.getValue();
641                        List<V> otherValue = map2.get(key);
642
643                        /*
644                         * The properties are stored in a ListMultimap. This class never
645                         * returns null. It returns an empty list when the key is not found.
646                         * But keep the null check just incase.
647                         */
648                        if (value == null) {
649                                if (otherValue != null || !map2.containsKey(key)) {
650                                        return false;
651                                }
652                                continue;
653                        }
654
655                        if (otherValue == null || value.size() != otherValue.size()) {
656                                return false;
657                        }
658
659                        List<V> otherValueCopy = new ArrayList<V>(otherValue);
660                        for (V property : value) {
661                                if (!otherValueCopy.remove(property)) {
662                                        return false;
663                                }
664                        }
665                }
666                return true;
667        }
