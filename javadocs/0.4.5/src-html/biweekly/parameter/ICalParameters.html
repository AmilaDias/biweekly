550        /**
551         * Checks this parameters list for data consistency problems or deviations
552         * from the spec. These problems will not prevent the iCalendar object from
553         * being written to a data stream, but may prevent it from being parsed
554         * correctly by the consuming application.
555         * @param version the version to validate against
556         * @return a list of warnings or an empty list if no problems were found
557         */
558        public List<Warning> validate(ICalVersion version) {
559                List<Warning> warnings = new ArrayList<Warning>(0);
560
561                /*
562                 * Check for invalid characters in names and values.
563                 */
564                {
565                        BitSet invalidValueChars = new BitSet(128);
566                        invalidValueChars.set(0, 31);
567                        invalidValueChars.set(127);
568                        invalidValueChars.set('\t', false); //allow
569                        invalidValueChars.set('\n', false); //allow
570                        invalidValueChars.set('\r', false); //allow
571                        if (version == ICalVersion.V1_0) {
572                                invalidValueChars.set(',');
573                                invalidValueChars.set('.');
574                                invalidValueChars.set(':');
575                                invalidValueChars.set('=');
576                                invalidValueChars.set('[');
577                                invalidValueChars.set(']');
578                        }
579
580                        CharacterBitSet validNameChars = new CharacterBitSet("-a-zA-Z0-9");
581                        for (Map.Entry<String, List<String>> entry : this) {
582                                String name = entry.getKey();
583
584                                //check the parameter name
585                                if (!validNameChars.containsOnly(name)) {
586                                        warnings.add(Warning.validate(54, name));
587                                }
588
589                                //check the parameter value(s)
590                                List<String> values = entry.getValue();
591                                for (String value : values) {
592                                        for (int i = 0; i < value.length(); i++) {
593                                                char c = value.charAt(i);
594                                                if (invalidValueChars.get(c)) {
595                                                        warnings.add(Warning.validate(53, name, value, (int) c, i));
596                                                        break;
597                                                }
598                                        }
599                                }
600                        }
601                }
602
603                final int nonStandardCode = 1, deprecated = 47;
604
605                String value = first(RSVP);
606                if (value != null) {
607                        value = value.toLowerCase();
608                        List<String> validValues = Arrays.asList("true", "false", "yes", "no");
609                        if (!validValues.contains(value)) {
610                                warnings.add(Warning.validate(nonStandardCode, RSVP, value, validValues));
611                        }
612                }
613
614                value = first(CUTYPE);
615                if (value != null && CalendarUserType.find(value) == null) {
616                        warnings.add(Warning.validate(nonStandardCode, CUTYPE, value, CalendarUserType.all()));
617                }
618
619                value = first(ENCODING);
620                if (value != null && Encoding.find(value) == null) {
621                        warnings.add(Warning.validate(nonStandardCode, ENCODING, value, Encoding.all()));
622                }
623
624                value = first(FBTYPE);
625                if (value != null && FreeBusyType.find(value) == null) {
626                        warnings.add(Warning.validate(nonStandardCode, FBTYPE, value, FreeBusyType.all()));
627                }
628
629                value = first(PARTSTAT);
630                if (value != null && ParticipationStatus.find(value) == null) {
631                        warnings.add(Warning.validate(nonStandardCode, PARTSTAT, value, ParticipationStatus.all()));
632                }
633
634                value = first(RANGE);
635                if (value != null) {
636                        Range range = Range.find(value);
637
638                        if (range == null) {
639                                warnings.add(Warning.validate(nonStandardCode, RANGE, value, Range.all()));
640                        }
641
642                        if (range == Range.THIS_AND_PRIOR && version == ICalVersion.V2_0) {
643                                warnings.add(Warning.validate(deprecated, RANGE, value));
644                        }
645                }
646
647                value = first(RELATED);
648                if (value != null && Related.find(value) == null) {
649                        warnings.add(Warning.validate(nonStandardCode, RELATED, value, Related.all()));
650                }
651
652                value = first(RELTYPE);
653                if (value != null && RelationshipType.find(value) == null) {
654                        warnings.add(Warning.validate(nonStandardCode, RELTYPE, value, RelationshipType.all()));
655                }
656
657                value = first(ROLE);
658                if (value != null && Role.find(value) == null) {
659                        warnings.add(Warning.validate(nonStandardCode, ROLE, value, Role.all()));
660                }
661
662                value = first(VALUE);
663                if (value != null && ICalDataType.find(value) == null) {
664                        warnings.add(Warning.validate(nonStandardCode, VALUE, value, ICalDataType.all()));
665                }
666
667                return warnings;
668        }

675        @Override
676        public int hashCode() {
677                /*
678                 * Remember: Keys are case-insensitive, key order does not matter, and
679                 * value order does not matter
680                 */
681                final int prime = 31;
682                int result = 1;
683
684                for (Map.Entry<String, List<String>> entry : this) {
685                        String key = entry.getKey();
686                        List<String> value = entry.getValue();
687
688                        int valueHash = 1;
689                        for (String v : value) {
690                                valueHash += v.toLowerCase().hashCode();
691                        }
692
693                        int entryHash = 1;
694                        entryHash += prime * entryHash + key.toLowerCase().hashCode();
695                        entryHash += prime * entryHash + valueHash;
696
697                        result += entryHash;
698                }
699
700                return result;
701        }

703        /**
704         * <p>
705         * Determines whether the given object is logically equivalent to this list
706         * of iCalendar parameters.
707         * </p>
708         * <p>
709         * iCalendar parameters are case-insensitive. Also, the order in which they
710         * are defined does not matter.
711         * </p>
712         * @param obj the object to compare to
713         * @return true if the objects are equal, false if not
714         */
715        @Override
716        public boolean equals(Object obj) {
717                /*
718                 * Remember: Keys are case-insensitive, key order does not matter, and
719                 * value order does not matter
720                 */
721                if (this == obj) return true;
722                if (obj == null) return false;
723                if (getClass() != obj.getClass()) return false;
724
725                ICalParameters other = (ICalParameters) obj;
726                if (size() != other.size()) return false;
727
728                for (Map.Entry<String, List<String>> entry : this) {
729                        String key = entry.getKey();
730                        List<String> value = entry.getValue();
731
732                        /*
733                         * The parameters are stored in a ListMultimap. This class never
734                         * returns null. It returns an empty list when the key is not found.
735                         * But keep the null check just incase.
736                         */
737                        if (value == null) {
738                                if (other.get(key) != null || !other.containsKey(key)) {
739                                        return false;
740                                }
741                                continue;
742                        }
743
744                        List<String> otherValue = other.get(key);
745                        if (otherValue == null || value.size() != otherValue.size()) {
746                                return false;
747                        }
748
749                        List<String> valueLower = new ArrayList<String>(value.size());
750                        for (String v : value) {
751                                valueLower.add(v.toLowerCase());
752                        }
753                        Collections.sort(valueLower);
754
755                        List<String> otherValueLower = new ArrayList<String>(otherValue.size());
756                        for (String v : otherValue) {
757                                otherValueLower.add(v.toLowerCase());
758                        }
759                        Collections.sort(otherValueLower);
760
761                        if (!valueLower.equals(otherValueLower)) {
762                                return false;
763                        }
764                }
765
766                return true;
767        }
