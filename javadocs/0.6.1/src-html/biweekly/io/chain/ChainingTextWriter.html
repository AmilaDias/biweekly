045/**
046 * Chainer class for writing traditional, plain-text iCalendar objects.
047 * @see Biweekly#write(Collection)
048 * @see Biweekly#write(ICalendar...)
049 * @author Michael Angstadt
050 */
051public class ChainingTextWriter extends ChainingWriter<ChainingTextWriter> {
052        private ICalVersion version;
053        private boolean caretEncoding = false;
054
055        /**
056         * @param icals the iCalendar objects to write
057         */
058        public ChainingTextWriter(Collection<ICalendar> icals) {
059                super(icals);
060        }
061
062        /**
063         * <p>
064         * Sets the version that all the iCalendar objects will be marshalled to.
065         * The version that is attached to each individual {@link ICalendar} object
066         * will be ignored.
067         * </p>
068         * <p>
069         * If no version is passed into this method, the writer will look at the
070         * version attached to each individual {@link ICalendar} object and marshal
071         * it to that version. And if a {@link ICalendar} object has no version
072         * attached to it, then it will be marshalled to version 2.0.
073         * </p>
074         * @param version the version to marshal the iCalendar objects to
075         * @return this
076         */
077        public ChainingTextWriter version(ICalVersion version) {
078                this.version = version;
079                return this;
080        }
081
082        /**
083         * Sets whether the writer will use circumflex accent encoding for parameter
084         * values (disabled by default). This only applies to version 2.0 iCalendar
085         * objects.
086         * @param enable true to use circumflex accent encoding, false not to
087         * @return this
088         * @see ICalWriter#setCaretEncodingEnabled(boolean)
089         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
090         */
091        public ChainingTextWriter caretEncoding(boolean enable) {
092                this.caretEncoding = enable;
093                return this;
094        }
095
096        @Override
097        public ChainingTextWriter tz(TimeZone defaultTimeZone, boolean outlookCompatible) {
098                return super.tz(defaultTimeZone, outlookCompatible);
099        }
100
101        @Override
102        public ChainingTextWriter register(ICalPropertyScribe<? extends ICalProperty> scribe) {
103                return super.register(scribe);
104        }
105
106        @Override
107        public ChainingTextWriter register(ICalComponentScribe<? extends ICalComponent> scribe) {
108                return super.register(scribe);
109        }
110
111        /**
112         * Writes the iCalendar objects to a string.
113         * @return the iCalendar string
114         */
115        public String go() {
116                StringWriter sw = new StringWriter();
117                try {
118                        go(sw);
119                } catch (IOException e) {
120                        //should never be thrown because we're writing to a string
121                        throw new RuntimeException(e);
122                }
123                return sw.toString();
124        }
125
126        /**
127         * Writes the iCalendar objects to an output stream.
128         * @param out the output stream to write to
129         * @throws IOException if there's a problem writing to the output stream
130         */
131        public void go(OutputStream out) throws IOException {
132                go(new ICalWriter(out, getICalWriterConstructorVersion()));
133        }
134
135        /**
136         * Writes the iCalendar objects to a file. If the file exists, it will be
137         * overwritten.
138         * @param file the file to write to
139         * @throws IOException if there's a problem writing to the file
140         */
141        public void go(File file) throws IOException {
142                go(file, false);
143        }
144
145        /**
146         * Writes the iCalendar objects to a file.
147         * @param file the file to write to
148         * @param append true to append onto the end of the file, false to overwrite
149         * it
150         * @throws IOException if there's a problem writing to the file
151         */
152        public void go(File file, boolean append) throws IOException {
153                ICalWriter writer = new ICalWriter(file, append, getICalWriterConstructorVersion());
154                try {
155                        go(writer);
156                } finally {
157                        writer.close();
158                }
159        }
160
161        /**
162         * Writes the iCalendar objects to a writer.
163         * @param writer the writer to write to
164         * @throws IOException if there's a problem writing to the writer
165         */
166        public void go(Writer writer) throws IOException {
167                go(new ICalWriter(writer, getICalWriterConstructorVersion()));
168        }
169
170        private void go(ICalWriter writer) throws IOException {
171                writer.setCaretEncodingEnabled(caretEncoding);
172                if (defaultTimeZone != null) {
173                        writer.setGlobalTimezone(defaultTimeZone);
174                }
175                if (index != null) {
176                        writer.setScribeIndex(index);
177                }
178
179                for (ICalendar ical : icals) {
180                        if (version == null) {
181                                //use the version that's assigned to each individual iCalendar object
182                                ICalVersion icalVersion = ical.getVersion();
183                                if (icalVersion == null) {
184                                        icalVersion = ICalVersion.V2_0;
185                                }
186                                writer.setTargetVersion(icalVersion);
187                        }
188                        writer.write(ical);
189                        writer.flush();
190                }
191        }
192
193        /**
194         * <p>
195         * Gets the {@link ICalVersion} object to pass into the {@link ICalWriter}
196         * constructor. The constructor does not allow a null version, so this
197         * method ensures that a non-null version is passed in.
198         * </p>
199         * <p>
200         * If the user hasn't chosen a version, the version that is passed into the
201         * constructor doesn't matter. This is because the writer's target version
202         * is reset every time an iCalendar object is written (see the
203         * {@link #go(ICalWriter)} method).
204         * </p>
205         * @return the version to pass into the constructor
206         */
207        private ICalVersion getICalWriterConstructorVersion() {
208                return (version == null) ? ICalVersion.V2_0 : version;
209        }
210}
