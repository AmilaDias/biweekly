015package biweekly.util.com.google.ical.compat.javautil;
016
017import biweekly.util.com.google.ical.iter.RecurrenceIterable;
018import biweekly.util.com.google.ical.iter.RecurrenceIterator;
019import biweekly.util.com.google.ical.iter.RecurrenceIteratorFactory;
020import biweekly.util.com.google.ical.util.TimeUtils;
021import biweekly.util.com.google.ical.values.DateTimeValueImpl;
022import biweekly.util.com.google.ical.values.DateValue;
023import biweekly.util.com.google.ical.values.DateValueImpl;
024import biweekly.util.com.google.ical.values.TimeValue;
025
026import java.text.ParseException;
027import java.util.Calendar;
028import java.util.Date;
029import java.util.GregorianCalendar;
030import java.util.TimeZone;
031
032/**
033 * a factory for converting RRULEs and RDATEs into
034 * <code>Iterator&lt;Date&gt;</code> and <code>Iterable&lt;Date&gt;</code>.
035 *
036 * @see RecurrenceIteratorFactory
037 *
038 * @author mikesamuel+svn@gmail.com (Mike Samuel)
039 */
040public class DateIteratorFactory {
041
042  /**
043   * given a block of RRULE, EXRULE, RDATE, and EXDATE content lines, parse
044   * them into a single date iterator.
045   * @param rdata RRULE, EXRULE, RDATE, and EXDATE lines.
046   * @param start the first occurrence of the series.
047   * @param tzid the local timezone -- used to interpret start and any dates in
048   *   RDATE and EXDATE lines that don't have TZID params.
049   * @param strict true if any failure to parse should result in a
050   *   ParseException.  false causes bad content lines to be logged and ignored.
051   */
052  public static DateIterator createDateIterator(
053      String rdata, Date start, TimeZone tzid, boolean strict)
054      throws ParseException {
055    return new RecurrenceIteratorWrapper(
056        RecurrenceIteratorFactory.createRecurrenceIterator(
057            rdata, dateToDateValue(start, true),
058            tzid, strict));
059  }
060
061  /**
062   * given a block of RRULE, EXRULE, RDATE, and EXDATE content lines, parse
063   * them into a single date iterable.
064   * @param rdata RRULE, EXRULE, RDATE, and EXDATE lines.
065   * @param start the first occurrence of the series.
066   * @param tzid the local timezone -- used to interpret start and any dates in
067   *   RDATE and EXDATE lines that don't have TZID params.
068   * @param strict true if any failure to parse should result in a
069   *   ParseException.  false causes bad content lines to be logged and ignored.
070   */
071  public static DateIterable createDateIterable(
072      String rdata, Date start, TimeZone tzid, boolean strict)
073      throws ParseException {
074    return new RecurrenceIterableWrapper(
075        RecurrenceIteratorFactory.createRecurrenceIterable(
076            rdata, dateToDateValue(start, true),
077            tzid, strict));
078  }
079
080  /**
081   * creates a date iterator given a recurrence iterator from
082   * {@link biweekly.util.com.google.ical.iter.RecurrenceIteratorFactory}.
083   */
084  public static DateIterator createDateIterator(RecurrenceIterator rit) {
085    return new RecurrenceIteratorWrapper(rit);
086  }
087
088  private static final class RecurrenceIterableWrapper
089      implements DateIterable {
090    private final RecurrenceIterable it;
091
092    public RecurrenceIterableWrapper(RecurrenceIterable it) { this.it = it; }
093
094    public DateIterator iterator() {
095      return new RecurrenceIteratorWrapper(it.iterator());
096    }
097  }
098
099  private static final class RecurrenceIteratorWrapper
100      implements DateIterator {
101    private final RecurrenceIterator it;
102    RecurrenceIteratorWrapper(RecurrenceIterator it) { this.it = it; }
103    public boolean hasNext() { return it.hasNext(); }
104    public Date next() { return dateValueToDate(it.next()); }
105    public void remove() { throw new UnsupportedOperationException(); }
106    public void advanceTo(Date d) {
107      // we need to treat midnight as a date value so that passing in
108      // dateValueToDate(<some-date-value>) will not advance past any
109      // occurrences of some-date-value in the iterator.
110      it.advanceTo(dateToDateValue(d, true));
111    }
112  }
113
114  static Date dateValueToDate(DateValue dvUtc) {
115    GregorianCalendar c = new GregorianCalendar(TimeUtils.utcTimezone());
116    c.clear();
117    if (dvUtc instanceof TimeValue) {
118      TimeValue tvUtc = (TimeValue) dvUtc;
119      c.set(dvUtc.year(),
120            dvUtc.month() - 1,  // java.util's dates are zero-indexed
121            dvUtc.day(),
122            tvUtc.hour(),
123            tvUtc.minute(),
124            tvUtc.second());
125    } else {
126      c.set(dvUtc.year(),
127            dvUtc.month() - 1,  // java.util's dates are zero-indexed
128            dvUtc.day(),
129            0,
130            0,
131            0);
132    }
133    return c.getTime();
134  }
135
136  static DateValue dateToDateValue(Date date, boolean midnightAsDate) {
137    GregorianCalendar c = new GregorianCalendar(TimeUtils.utcTimezone());
138    c.setTime(date);
139    int h = c.get(Calendar.HOUR_OF_DAY),
140      m = c.get(Calendar.MINUTE),
141      s = c.get(Calendar.SECOND);
142    if (midnightAsDate && 0 == (h | m | s)) {
143      return new DateValueImpl(c.get(Calendar.YEAR),
144                               c.get(Calendar.MONTH) + 1,
145                               c.get(Calendar.DAY_OF_MONTH));
146    } else {
147      return new DateTimeValueImpl(c.get(Calendar.YEAR),
148                                   c.get(Calendar.MONTH) + 1,
149                                   c.get(Calendar.DAY_OF_MONTH),
150                                   h,
151                                   m,
152                                   s);
153    }
154  }
155
156  private DateIteratorFactory() {
157    // uninstantiable
158  }
159}
