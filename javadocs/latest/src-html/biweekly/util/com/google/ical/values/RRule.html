022/**
023 * encapsulates an RFC 2445 RRULE or EXRULE.
024 *
025 * @author mikesamuel+svn@gmail.com (Mike Samuel)
026 */
027public class RRule extends AbstractIcalObject {
028
029  private Frequency freq;
030  private Weekday wkst;
031  private DateValue until;
032  private int count;
033  private int interval;
034  private List<WeekdayNum> byDay = new ArrayList<WeekdayNum>();
035  private int[] byMonth = NO_INTS;  // in +/-[1-12]
036  private int[] byMonthDay = NO_INTS;  // in +/-[1-31]
037  private int[] byWeekNo = NO_INTS;  // in +/-[1-53]
038  private int[] byYearDay = NO_INTS;  // in +/-[1-366]
039  private int[] byHour = NO_INTS;  // in [0-23]
040  private int[] byMinute = NO_INTS;  // in [0-59]
041  private int[] bySecond = NO_INTS;  // in [0-60]
042  private int[] bySetPos = NO_INTS;  // in +/-[1-366]
043
044  private static final int[] NO_INTS = new int[0];
045
046  public RRule() {
047    this.freq = Frequency.DAILY;
048    setName("RRULE");
049  }
050
051  public RRule(String icalString) throws ParseException {
052    parse(VcalRewriter.rewriteRule(icalString), RRuleSchema.instance());
053  }

055  /**
056   * formats as an *unfolded* RFC 2445 content line.
057   */
058  public String toIcal() {
059    StringBuilder buf = new StringBuilder();
060    buf.append(this.getName());
061    if (hasExtParams()) {
062      for (Map.Entry<String, String> param : getExtParams().entrySet()) {
063        String k = param.getKey(),
064               v = param.getValue();
065        if (ICAL_SPECIALS.matcher(v).find()) {
066          v = "\"" + v + "\"";
067        }
068        buf.append(';').append(k).append('=').append(v);
069      }
070    }
071    buf.append(":FREQ=").append(freq);
072    if (null != wkst) {
073      buf.append(";WKST=").append(wkst.toString());
074    }
075    if (null != this.until) {
076      buf.append(";UNTIL=").append(until);
077      if (until instanceof TimeValue) {
078        buf.append('Z');
079      }
080    }
081    if (count != 0) {
082      buf.append(";COUNT=").append(count);
083    }
084    if (interval != 0) {
085      buf.append(";INTERVAL=").append(interval);
086    }
087    if (0 != byYearDay.length) {
088      buf.append(";BYYEARDAY=");
089      writeIntList(byYearDay, buf);
090    }
091    if (0 != byMonth.length) {
092      buf.append(";BYMONTH=");
093      writeIntList(byMonth, buf);
094    }
095    if (0 != byMonthDay.length) {
096      buf.append(";BYMONTHDAY=");
097      writeIntList(byMonthDay, buf);
098    }
099    if (0 != byWeekNo.length) {
100      buf.append(";BYWEEKNO=");
101      writeIntList(byWeekNo, buf);
102    }
103    if (!byDay.isEmpty()) {
104      buf.append(";BYDAY=");
105      boolean first = true;
106      for (WeekdayNum day : this.byDay) {
107        if (!first) {
108          buf.append(',');
109        } else {
110          first = false;
111        }
112        buf.append(day);
113      }
114    }
115    if (0 != byHour.length) {
116      buf.append(";BYHOUR=");
117      writeIntList(byHour, buf);
118    }
119    if (0 != byMinute.length) {
120      buf.append(";BYMINUTE=");
121      writeIntList(byMinute, buf);
122    }
123    if (0 != bySecond.length) {
124      buf.append(";BYSECOND=");
125      writeIntList(bySecond, buf);
126    }
127    if (0 != bySetPos.length) {
128      buf.append(";BYSETPOS=");
129      writeIntList(bySetPos, buf);
130    }
131    return buf.toString();
132  }
133
134  private static void writeIntList(int[] nums, StringBuilder out) {
135    for (int i = 0; i < nums.length; ++i) {
136      if (0 != i) { out.append(','); }
137      out.append(nums[i]);
138    }
139  }

141  /** an approximate number of days between occurences. */
142  public int approximateIntervalInDays() {
143    int freqLengthDays;
144    int nPerPeriod = 0;
145    switch (this.freq) {
146      case DAILY:
147        freqLengthDays = 1;
148        break;
149      case WEEKLY:
150        freqLengthDays = 7;
151        if (!this.byDay.isEmpty()) {
152          nPerPeriod = this.byDay.size();
153        }
154        break;
155      case MONTHLY:
156        freqLengthDays = 30;
157        if (!this.byDay.isEmpty()) {
158          for (WeekdayNum day : byDay) {
159            // if it's every weekday in the month, assume four of that weekday,
160            // otherwise there is one of that week-in-month,weekday pair
161            nPerPeriod += 0 != day.num ? 1 : 4;
162          }
163        } else {
164          nPerPeriod = this.byMonthDay.length;
165        }
166        break;
167      case YEARLY:
168        freqLengthDays = 365;
169
170        int monthCount = 12;
171        if (0 != this.byMonth.length) {
172          monthCount = this.byMonth.length;
173        }
174
175        if (!this.byDay.isEmpty()) {
176          for (WeekdayNum day : byDay) {
177            // if it's every weekend in the months in the year,
178            // assume 4 of that weekday per month,
179            // otherwise there is one of that week-in-month,weekday pair per
180            // month
181            nPerPeriod += (0 != day.num ? 1 : 4) * monthCount;
182          }
183        } else if (0 != this.byMonthDay.length) {
184          nPerPeriod += monthCount * this.byMonthDay.length;
185        } else {
186          nPerPeriod += this.byYearDay.length;
187        }
188        break;
189      default: freqLengthDays = 0;
190    }
191    if (0 == nPerPeriod) { nPerPeriod = 1; }
192
193    return ((freqLengthDays / nPerPeriod) * this.interval);
194  }
