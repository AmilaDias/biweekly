018package biweekly.util.org.apache.commons.codec.binary;
019
020import java.io.FilterInputStream;
021import java.io.IOException;
022import java.io.InputStream;
023
024/**
025 * Abstract superclass for Base-N input streams.
026 * 
027 * @since 1.5
028 */
029public class BaseNCodecInputStream extends FilterInputStream {
030
031    private final boolean doEncode;
032
033    private final BaseNCodec baseNCodec;
034
035    private final byte[] singleByte = new byte[1];
036
037    protected BaseNCodecInputStream(InputStream in, BaseNCodec baseNCodec, boolean doEncode) {
038        super(in);
039        this.doEncode = doEncode;
040        this.baseNCodec = baseNCodec;
041    }
042
043    /**
044     * Reads one <code>byte</code> from this input stream.
045     * 
046     * @return the byte as an integer in the range 0 to 255. Returns -1 if EOF has been reached.
047     * @throws IOException
048     *             if an I/O error occurs.
049     */
050    @Override
051    public int read() throws IOException {
052        int r = read(singleByte, 0, 1);
053        while (r == 0) {
054            r = read(singleByte, 0, 1);
055        }
056        if (r > 0) {
057            return singleByte[0] < 0 ? 256 + singleByte[0] : singleByte[0];
058        }
059        return -1;
060    }
061
062    /**
063     * Attempts to read <code>len</code> bytes into the specified <code>b</code> array starting at <code>offset</code>
064     * from this InputStream.
065     * 
066     * @param b
067     *            destination byte array
068     * @param offset
069     *            where to start writing the bytes
070     * @param len
071     *            maximum number of bytes to read
072     * 
073     * @return number of bytes read
074     * @throws IOException
075     *             if an I/O error occurs.
076     * @throws NullPointerException
077     *             if the byte array parameter is null
078     * @throws IndexOutOfBoundsException
079     *             if offset, len or buffer size are invalid
080     */
081    @Override
082    public int read(byte b[], int offset, int len) throws IOException {
083        if (b == null) {
084            throw new NullPointerException();
085        } else if (offset < 0 || len < 0) {
086            throw new IndexOutOfBoundsException();
087        } else if (offset > b.length || offset + len > b.length) {
088            throw new IndexOutOfBoundsException();
089        } else if (len == 0) {
090            return 0;
091        } else {
092            int readLen = 0;
093            /*
094             Rationale for while-loop on (readLen == 0):
095             -----
096             Base32.readResults() usually returns > 0 or EOF (-1).  In the
097             rare case where it returns 0, we just keep trying.
098
099             This is essentially an undocumented contract for InputStream
100             implementors that want their code to work properly with
101             java.io.InputStreamReader, since the latter hates it when
102             InputStream.read(byte[]) returns a zero.  Unfortunately our
103             readResults() call must return 0 if a large amount of the data
104             being decoded was non-base32, so this while-loop enables proper
105             interop with InputStreamReader for that scenario.
106             -----
107             This is a fix for CODEC-101
108            */
109            while (readLen == 0) {
110                if (!baseNCodec.hasData()) {
111                    byte[] buf = new byte[doEncode ? 4096 : 8192];
112                    int c = in.read(buf);
113                    if (doEncode) {
114                        baseNCodec.encode(buf, 0, c);
115                    } else {
116                        baseNCodec.decode(buf, 0, c);
117                    }
118                }
119                readLen = baseNCodec.readResults(b, offset, len);
120            }
121            return readLen;
122        }
123    }
124    /**
125     * {@inheritDoc}
126     * 
127     * @return false
128     */
129    @Override
130    public boolean markSupported() {
131        return false; // not an easy job to support marks
132    }
133
134}
