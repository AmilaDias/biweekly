045  /**
046   * Get a "time_t" in millis given a number of seconds since
047   * Dershowitz/Reingold epoch relative to a given timezone.
048   * @param epochSecs Number of seconds since Dershowitz/Reingold
049   * epoch, relatve to zone.
050   * @param zone Timezone against which epochSecs applies
051   * @return Number of milliseconds since 00:00:00 Jan 1, 1970 GMT
052   */
053  private static long timetMillisFromEpochSecs(long epochSecs,
054                                               TimeZone zone) {
055    DateTimeValue date = timeFromSecsSinceEpoch(epochSecs);
056    Calendar cal = new GregorianCalendar(zone);
057    cal.clear(); // clear millis
058    cal.setTimeZone(zone);
059    cal.set(date.year(), date.month() - 1, date.day(),
060            date.hour(), date.minute(), date.second());
061    return cal.getTimeInMillis();
062  }

064  private static DateTimeValue convert(DateTimeValue time,
065                                       TimeZone zone,
066                                       int sense) {
067    if (zone == null ||
068        zone.hasSameRules(ZULU) ||
069        time.year() == 0) {
070      return time;
071    }
072
073    long timetMillis;
074
075    if (sense > 0) {
076      // time is in UTC
077      timetMillis = timetMillisFromEpochSecs(secsSinceEpoch(time), ZULU);
078    } else {
079      // time is in local time; since zone.getOffset() expects millis
080      // in UTC, need to convert before we can get the offset (ironic)
081      timetMillis = timetMillisFromEpochSecs(secsSinceEpoch(time), zone);
082    }
083
084    int millisecondOffset = zone.getOffset(timetMillis);
085    int millisecondRound = millisecondOffset < 0 ? -500 : 500;
086    int secondOffset = (millisecondOffset + millisecondRound) / 1000;
087
088    DateTimeValue dtv = toDateTimeValue(timetMillis, zone);
089    return addSeconds(dtv, sense * secondOffset);
090  }

108  private static DateTimeValue addSeconds(DateTimeValue dtime, int seconds) {
109    return new DTBuilder(dtime.year(), dtime.month(),
110                         dtime.day(), dtime.hour(),
111                         dtime.minute(),
112                         dtime.second() + seconds).toDateTime();
113  }

115  public static DateValue add(DateValue d, DateValue dur) {
116    DTBuilder db = new DTBuilder(d);
117    db.year += dur.year();
118    db.month += dur.month();
119    db.day += dur.day();
120    if (dur instanceof TimeValue) {
121      TimeValue tdur = (TimeValue) dur;
122      db.hour += tdur.hour();
123      db.minute += tdur.minute();
124      db.second += tdur.second();
125      return db.toDateTime();
126    } else if (d instanceof TimeValue) {
127      return db.toDateTime();
128    }
129    return db.toDate();
130  }

151  private static int fixedFromGregorian(DateValue date) {
152    return fixedFromGregorian(date.year(), date.month(), date.day());
153  }

155  /**
156   * the number of days since the <em>epoch</em>,
157   * which is the imaginary beginning of year zero in a hypothetical
158   * backward extension of the Gregorian calendar through time.
159   * See "Calendrical Calculations" by Reingold and Dershowitz.
160   */
161  public static int fixedFromGregorian(int year, int month, int day) {
162    int yearM1 = year - 1;
163    return 365 * yearM1 +
164      yearM1/4 -
165      yearM1/100 +
166      yearM1/400 +
167      (367*month - 362)/12 +
168      (month <= 2 ? 0 :
169       isLeapYear(year) ? -1 :
170       -2) +
171      day;
172  }
173
174  public static boolean isLeapYear(int year) {
175    return (year % 4 == 0) && ((year % 100 != 0) || (year % 400 == 0));
176  }
177
178  /** count of days inthe given year */
179  public static int yearLength(int year) {
180    return isLeapYear(year) ? 366 : 365;
181  }
182
183  /** count of days in the given month (one indexed) of the given year. */
184  public static int monthLength(int year, int month) {
185    switch (month) {
186      case 1:
187      case 3:
188      case 5:
189      case 7:
190      case 8:
191      case 10:
192      case 12:
193        return 31;
194      case 4:
195      case 6:
196      case 9:
197      case 11:
198        return 30;
199      case 2:
200        return isLeapYear(year) ? 29 : 28;
201      default:
202        throw new AssertionError(month);
203    }
204  }

206  private static int[] MONTH_START_TO_DOY = new int[12];
207  static {
208    assert !isLeapYear(1970);
209    for (int m = 1; m < 12; ++m) {
210      MONTH_START_TO_DOY[m] = MONTH_START_TO_DOY[m - 1] + monthLength(1970, m);
211    }
212    assert 365 == MONTH_START_TO_DOY[11] + monthLength(1970, 12) :
213           "" + (MONTH_START_TO_DOY[11] + monthLength(1970, 12));
214  }

216  /** the day of the year in [0-365] of the given date. */
217  public static int dayOfYear(int year, int month, int date) {
218    int leapAdjust = month > 2 && isLeapYear(year) ? 1 : 0;
219    return MONTH_START_TO_DOY[month - 1] + leapAdjust + date - 1;
220  }

222  /**
223   * Compute the gregorian time from the number of seconds since the
224   * Proleptic Gregorian Epoch.
225   * See "Calendrical Calculations", Reingold and Dershowitz.
226   */
227  public static DateTimeValue timeFromSecsSinceEpoch(long secsSinceEpoch) {
228    // TODO: should we handle -ve years?
229    int secsInDay = (int) (secsSinceEpoch % SECS_PER_DAY);
230    int daysSinceEpoch = (int) (secsSinceEpoch / SECS_PER_DAY);
231    int approx = (int) ((daysSinceEpoch + 10) * 400L / 146097);
232    int year = (daysSinceEpoch >= fixedFromGregorian(approx+1, 1, 1))
233               ? approx+1 : approx;
234    int jan1 = fixedFromGregorian(year, 1, 1);
235    int priorDays = daysSinceEpoch - jan1;
236    int march1 = fixedFromGregorian(year, 3, 1);
237    int correction = (daysSinceEpoch < march1) ? 0 :
238                     isLeapYear(year) ? 1 : 2;
239    int month = (12 * (priorDays + correction) + 373) / 367;
240    int month1 = fixedFromGregorian(year, month, 1);
241    int day = daysSinceEpoch - month1 + 1;
242    int second = secsInDay % 60;
243    int minutesInDay = secsInDay / 60;
244    int minute = minutesInDay % 60;
245    int hour = minutesInDay / 60;
246    if (!(hour >= 0 && hour < 24)) throw new AssertionError(
247        "Input was: " + secsSinceEpoch + "to make hour: " + hour);
248    DateTimeValue result =
249      new DateTimeValueImpl(year, month, day, hour, minute, second);
250    // assert result.equals(normalize(result));
251    // assert secsSinceEpoch(result) == secsSinceEpoch;
252    return result;
253  }
254
255  private static final long SECS_PER_DAY = 60L * 60 * 24;
256
257  /**
258   * Compute the number of seconds from the Proleptic Gregorian epoch
259   * to the given time.
260   */
261  public static long secsSinceEpoch(DateValue date) {
262    long result = fixedFromGregorian(date) *
263                  SECS_PER_DAY;
264    if (date instanceof TimeValue) {
265      TimeValue time = (TimeValue) date;
266      result +=
267        time.second() +
268        60 * (time.minute() +
269              60 * time.hour());
270    }
271    return result;
272  }

290  private static final Pattern UTC_TZID =
291      Pattern.compile("^GMT([+-]0(:00)?)?$|UTC|Zulu|Etc\\/GMT|Greenwich.*",
292                      Pattern.CASE_INSENSITIVE);

294  /**
295   * returns the timezone with the given name or null if no such timezone.
296   * calendar/common/ICalUtil uses this function
297   */
298  public static TimeZone timeZoneForName(String tzString) {
299    // This is a horrible hack since there is no easier way to get a timezone
300    // only if the string is recognized as a timezone.
301    // The TimeZone.getTimeZone javadoc says the following:
302    //   Returns:
303    //       the specified TimeZone, or the GMT zone if the given ID cannot be
304    //       understood.
305    TimeZone tz = TimeZone.getTimeZone(tzString);
306    if (tz.hasSameRules(BOGUS_TIMEZONE)) {
307      // see if the user really was asking for GMT because if
308      // TimeZone.getTimeZone can't recognize tzString, then that is what it
309      // will return.
310      Matcher m = UTC_TZID.matcher(tzString);
311      if (m.matches()) {
312        return TimeUtils.utcTimezone();
313      }
314      // unrecognizable timezone
315      return null;
316    }
317    return tz;
318  }

320  public static DateTimeValue toDateTimeValue(long millisFromEpoch, TimeZone zone) {
321    GregorianCalendar c = new GregorianCalendar(zone);
322    c.clear();
323    c.setTimeInMillis(millisFromEpoch);
324    return new DateTimeValueImpl (
325            c.get(Calendar.YEAR),
326            c.get(Calendar.MONTH) + 1,
327            c.get(Calendar.DAY_OF_MONTH),
328            c.get(Calendar.HOUR_OF_DAY),
329            c.get(Calendar.MINUTE),
330            c.get(Calendar.SECOND));
331  }
