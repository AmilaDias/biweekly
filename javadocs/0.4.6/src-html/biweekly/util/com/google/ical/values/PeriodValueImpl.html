015package biweekly.util.com.google.ical.values;
016
017import biweekly.util.com.google.ical.util.TimeUtils;
018
019/**
020 * a half-open range of {@link DateValue}s.
021 *
022 * @author mikesamuel+svn@gmail.com (Mike Samuel)
023 */
024public class PeriodValueImpl implements PeriodValue {
025
026  private DateValue start, end;
027
028  /**
029   * returns a period with the given start and end dates.
030   * @param start non null.
031   * @param end on or after start.  May/must have a time if the start has a
032   *   time.
033   */
034  public static PeriodValue create(DateValue start, DateValue end) {
035    return new PeriodValueImpl(start, end);
036  }
037
038  /**
039   * returns a period with the given start date and duration.
040   * @param start non null.
041   * @param dur a positive duration represented as a DateValue.
042   */
043  public static PeriodValue createFromDuration(DateValue start, DateValue dur) {
044    DateValue end = TimeUtils.add(start, dur);
045    if (end instanceof TimeValue && !(start instanceof TimeValue)) {
046      start = TimeUtils.dayStart(start);
047    }
048    return new PeriodValueImpl(start, end);
049  }
050
051  protected PeriodValueImpl(DateValue start, DateValue end) {
052    if (start.compareTo(end) > 0) {
053      throw new IllegalArgumentException
054          ("Start (" + start + ") must precede end (" + end + ")");
055    }
056    if ((start instanceof TimeValue) ^ (end instanceof TimeValue)) {
057      throw new IllegalArgumentException
058        ("Start (" + start + ") and end (" + end +
059         ") must both have times or neither have times.");
060    }
061    this.start = start;
062    this.end = end;
063  }
064
065  public DateValue start() { return start; }
066
067  public DateValue end() { return end; }
068
069  /** true iff this period overlaps the given period. */
070  public boolean intersects(PeriodValue pv) {
071    DateValue sa = this.start,
072              ea = this.end,
073              sb = pv.start(),
074              eb = pv.end();
075
076    return sa.compareTo(eb) < 0 && sb.compareTo(ea) < 0;
077  }
078
079  /** true iff this period completely contains the given period. */
080  public boolean contains(PeriodValue pv) {
081    DateValue sa = this.start,
082              ea = this.end,
083              sb = pv.start(),
084              eb = pv.end();
085
086    return !(sb.compareTo(sa) < 0 || ea.compareTo(eb) < 0);
087  }
088
089  @Override public boolean equals(Object o) {
090    if (!(o instanceof PeriodValue)) { return false; }
091    PeriodValue that = (PeriodValue) o;
092    return this.start().equals(that.start())
093        && this.end().equals(that.end());
094  }
095
096  @Override public int hashCode() {
097    return start.hashCode() ^ (31 * end.hashCode());
098  }
099
100  @Override public String toString() {
101    return start().toString() + "/" + end().toString();
102  }
103
104}  // PeriodValueImpl
