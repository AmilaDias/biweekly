015package biweekly.util.com.google.ical.util;
016
017import biweekly.util.com.google.ical.util.TimeUtils;
018import biweekly.util.com.google.ical.values.DateTimeValue;
019import biweekly.util.com.google.ical.values.DateTimeValueImpl;
020import biweekly.util.com.google.ical.values.DateValue;
021import biweekly.util.com.google.ical.values.DateValueImpl;
022import biweekly.util.com.google.ical.values.TimeValue;
023
024/**
025 * a mutable buffer that can be used to build {@link DateValue}s and
026 * {@link DateTimeValue}s.
027 *
028 * @author mikesamuel+svn@gmail.com (Mike Samuel)
029 */
030public class DTBuilder {
031
032  /** in AD.  0 -> 1BC. */
033  public int year;
034  /** one indexed. */
035  public int month;
036  /** one indexed */
037  public int day;
038  /** zero indexed in 24 hour time. */
039  public int hour;
040  /** zero indexed */
041  public int minute;
042  /** zero indexed */
043  public int second;
044
045  public DTBuilder(int year, int month, int day,
046                   int hour, int minute, int second) {
047    this.year = year;
048    this.month = month;
049    this.day = day;
050    this.hour = hour;
051    this.minute = minute;
052    this.second = second;
053  }
054
055  public DTBuilder(int year, int month, int day) {
056    this.year = year;
057    this.month = month;
058    this.day = day;
059  }
060
061  public DTBuilder(DateValue dv) {
062    this.year = dv.year();
063    this.month = dv.month();
064    this.day = dv.day();
065    if (dv instanceof TimeValue) {
066      TimeValue tv = (TimeValue) dv;
067      this.hour = tv.hour();
068      this.minute = tv.minute();
069      this.second = tv.second();
070    }
071  }
072
073  /**
074   * produces a normalized date time, using zero for the time fields if none
075   * were provided.
076   * @return not null
077   */
078  public DateTimeValue toDateTime() {
079    normalize();
080    return new DateTimeValueImpl(year, month, day, hour, minute, second);
081  }
082
083  /**
084   * produces a normalized date.
085   * @return not null
086   */
087  public DateValue toDate() {
088    normalize();
089    return new DateValueImpl(year, month, day);
090  }
091
092  /**
093   * behavior undefined unless normalized.
094   * If you're not sure whether it's appropriate to use this method, use
095   * <code>toDateValue().compareTo(dv)</code> instead.
096   */
097  public int compareTo(DateValue dv) {
098    long dvComparable =
099      (((((long) dv.year()) << 4) + dv.month()) << 5) + dv.day();
100    long dtbComparable =
101      ((((long) year << 4) + month << 5)) + day;
102    if (dv instanceof TimeValue) {
103      TimeValue tv = (TimeValue) dv;
104      dvComparable = (((((dvComparable << 5) + tv.hour()) << 6) + tv.minute())
105                      << 6) + tv.second();
106      dtbComparable = (((((dtbComparable << 5) + hour) << 6) + minute)
107                       << 6) + second;
108    }
109    long delta = dtbComparable - dvComparable;
110    return delta < 0 ? -1 : delta == 0 ? 0 : 1;
111  }
112
113  /**
114   * makes sure that the fields are in the proper ranges, by e.g. converting
115   * 32 January to 1 February, or month 0 to December of the year before.
116   */
117  public void normalize() {
118    this.normalizeTime();
119    this.normalizeDate();
120  }
121
122  @Override
123  public String toString() {
124    return year + "-" + month + "-" + day + " " + hour + ":" + minute + ":"
125      + second;
126  }
127
128  @Override
129  public boolean equals(Object o) {
130    if (!(o instanceof DTBuilder)) { return false; }
131    DTBuilder that = (DTBuilder) o;
132    return this.year == that.year
133       && this.month == that.month
134         && this.day == that.day
135        && this.hour == that.hour
136      && this.minute == that.minute
137      && this.second == that.second;
138  }
139
140  @Override
141  public int hashCode() {
142    return
143      ((((((((year << 4) + month << 5) + day) << 5) + hour) << 6) + minute)
144       << 6) + second;
145  }
146
147  private void normalizeTime() {
148    int addMinutes = ((second < 0) ? (second - 59) : second) / 60;
149    second -= addMinutes * 60;
150    minute += addMinutes;
151    int addHours = ((minute < 0) ? (minute - 59) : minute) / 60;
152    minute -= addHours * 60;
153    hour += addHours;
154    int addDays = ((hour < 0) ? (hour - 23) : hour) / 24;
155    hour -= addDays * 24;
156    day += addDays;
157  }
158  private void normalizeDate() {
159    while (day <= 0) {
160      int days = TimeUtils.yearLength(month > 2 ? year : year - 1);
161      day += days;
162      --year;
163    }
164    if (month <= 0) {
165      int years = month / 12 - 1;
166      year += years;
167      month -= 12 * years;
168    } else if (month > 12) {
169      int years = (month - 1) / 12;
170      year += years;
171      month -= 12 * years;
172    }
173    while (true) {
174      if (month == 1) {
175        int yearLength = TimeUtils.yearLength(year);
176        if (day > yearLength) {
177          ++year;
178          day -= yearLength;
179        }
180      }
181      int monthLength = TimeUtils.monthLength(year, month);
182      if (day > monthLength) {
183        day -= monthLength;
184        if (++month > 12) {
185          month -= 12;
186          ++year;
187        }
188      } else {
189        break;
190      }
191    }
192  }
193
194}
