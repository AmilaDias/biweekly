003package biweekly.util.com.google.ical.util;
004
005import java.io.Serializable;
006
007/**
008 * A function with a boolean return value useful for filtering.
009 */
010public interface Predicate<T> extends Serializable {
011
012  /**
013   * Applies this Predicate to the given object.
014   *
015   * @param t may be null.
016   * @return the value of this Predicate when applied to input {@code t}
017   */
018  boolean apply(T t);
019}
