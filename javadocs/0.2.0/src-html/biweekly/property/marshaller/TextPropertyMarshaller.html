035    /**
036     * Marshals properties that have text values.
037     * @author Michael Angstadt
038     */
039    public abstract class TextPropertyMarshaller<T extends TextProperty> extends ICalPropertyMarshaller<T> {
040            protected final Value dataType;
041    
042            public TextPropertyMarshaller(Class<T> clazz, String propertyName) {
043                    this(clazz, propertyName, Value.TEXT);
044            }
045    
046            public TextPropertyMarshaller(Class<T> clazz, String propertyName, Value dataType) {
047                    super(clazz, propertyName);
048                    this.dataType = dataType;
049            }
050    
051            @Override
052            protected String _writeText(T property) {
053                    String value = property.getValue();
054                    if (value == null) {
055                            value = "";
056                    }
057                    return escape(value);
058            }
059    
060            @Override
061            protected T _parseText(String value, ICalParameters parameters, List<String> warnings) {
062                    value = unescape(value);
063                    return newInstance(value);
064            }
065    
066            @Override
067            protected void _writeXml(T property, XCalElement element) {
068                    element.append(dataType, property.getValue());
069            }
070    
071            @Override
072            protected T _parseXml(XCalElement element, ICalParameters parameters, List<String> warnings) {
073                    return newInstance(element.first(dataType));
074            }
075    
076            protected abstract T newInstance(String value);
077    }
