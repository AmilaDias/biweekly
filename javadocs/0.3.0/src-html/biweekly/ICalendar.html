050    /**
051     * <p>
052     * Represents an iCalendar object.
053     * </p>
054     * 
055     * <p>
056     * <b>Examples:</b>
057     * 
058     * <pre>
059     * ICalendar ical = new ICalendar();
060     * 
061     * VEvent event = new VEvent();
062     * event.setSummary("Team Meeting");
063     * Date start = ...;
064     * event.setDateStart(start);
065     * Date end = ...;
066     * event.setDateEnd(end);
067     * ical.addEvent(event);
068     * </pre>
069     * 
070     * </p>
071     * @author Michael Angstadt
072     * @see <a href="http://tools.ietf.org/html/rfc5545">RFC 5545</a>
073     */

075            /**
076             * <p>
077             * Creates a new iCalendar object.
078             * </p>
079             * <p>
080             * The following properties are auto-generated on object creation. These
081             * properties <b>must</b> be present in order for the iCalendar object to be
082             * valid:
083             * <ul>
084             * <li>{@link Version} - Set to the default iCalendar version ("2.0").</li>
085             * <li>{@link ProductId} - Set to a value that represents this library.</li>
086             * </ul>
087             * </p>
088             */
089            public ICalendar() {
090                    setVersion(Version.v2_0());
091                    setProductId(ProductId.biweekly());
092            }

298            /**
299             * Checks this iCalendar object for data consistency problems or deviations
300             * from the spec. These problems will not prevent the iCalendar object from
301             * being written to a data stream, but may prevent it from being parsed
302             * correctly by the consuming application. These problems can largely be
303             * avoided by reading the Javadocs of the component and property classes, or
304             * by being familiar with the iCalendar standard.
305             * @return the validation warnings
306             */
307            public ValidationWarnings validate() {
308                    //TODO make concurrent
309                    List<WarningsGroup> warnings = validate(new ArrayList<ICalComponent>(0));
310                    return new ValidationWarnings(warnings);
311            }
312    
313            @SuppressWarnings("unchecked")
314            @Override
315            protected void validate(List<ICalComponent> components, List<String> warnings) {
316                    checkRequiredCardinality(warnings, ProductId.class, Version.class);
317    
318                    if (this.components.isEmpty()) {
319                            warnings.add("An iCalendar object must have at least one component.");
320                    }
321            }
