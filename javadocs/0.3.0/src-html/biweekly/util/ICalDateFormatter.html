035    /**
036     * Helper class that formats and parses iCalendar dates. iCalendar dates adhere
037     * to the ISO8601 date format standard.
038     * @author Michael Angstadt
039     */
040    public class ICalDateFormatter {
041            /**
042             * Regular expression used to parse timezone offset strings.
043             */
044            private static final Pattern timeZoneRegex = Pattern.compile("^([-\\+])?(\\d{1,2})(:?(\\d{2}))?$");
045    
046            /**
047             * Formats a date for inclusion in an iCalendar object.
048             * @param date the date to format
049             * @param format the format to use
050             * @return the formatted date
051             */
052            public static String format(Date date, ISOFormat format) {
053                    return format(date, format, null);
054            }
055    
056            /**
057             * Formats a date for inclusion in an iCalendar object.
058             * @param date the date to format
059             * @param format the format to use
060             * @param timeZone the timezone to format the date in or null to use the
061             * JVM's default timezone (ignored with "UTC" formats)
062             * @return the formatted date
063             */
064            public static String format(Date date, ISOFormat format, TimeZone timeZone) {
065                    switch (format) {
066                    case UTC_TIME_BASIC:
067                    case UTC_TIME_EXTENDED:
068                            timeZone = TimeZone.getTimeZone("UTC");
069                            break;
070                    }
071    
072                    DateFormat df = format.getFormatDateFormat();
073                    if (timeZone != null) {
074                            df.setTimeZone(timeZone);
075                    }
076                    String str = df.format(date);
077    
078                    switch (format) {
079                    case TIME_EXTENDED:
080                            //add a colon to the timezone
081                            //example: converts "2012-07-05T22:31:41-0400" to "2012-07-05T22:31:41-04:00"
082                            str = str.replaceAll("([-\\+]\\d{2})(\\d{2})$", "$1:$2");
083                            break;
084                    }
085    
086                    return str;
087            }
088    
089            /**
090             * Parses an iCalendar date.
091             * @param dateStr the date string to parse (e.g. "20130609T181023Z")
092             * @return the parsed date
093             * @throws IllegalArgumentException if the date string isn't in one of the
094             * accepted ISO8601 formats
095             */
096            public static Date parse(String dateStr) {
097                    return parse(dateStr, null);
098            }
099    
100            /**
101             * Parses an iCalendar date.
102             * @param dateStr the date string to parse (e.g. "20130609T181023Z")
103             * @param timezone the timezone to parse the date as or null to use the
104             * JVM's default timezone (if the date string contains its own timezone,
105             * then that timezone will be used instead)
106             * @return the parsed date
107             * @throws IllegalArgumentException if the date string isn't in one of the
108             * accepted ISO8601 formats
109             */
110            public static Date parse(String dateStr, TimeZone timezone) {
111                    //find out what ISOFormat the date is in
112                    ISOFormat format = null;
113                    for (ISOFormat f : ISOFormat.values()) {
114                            if (f.matches(dateStr)) {
115                                    format = f;
116                                    break;
117                            }
118                    }
119                    if (format == null) {
120                            throw new IllegalArgumentException("Date string is not in a valid ISO-8601 format.");
121                    }
122    
123                    //tweak the date string to make it work with SimpleDateFormat
124                    switch (format) {
125                    case TIME_EXTENDED:
126                    case HCARD_TIME_TAG:
127                            //SimpleDateFormat doesn't recognize timezone offsets that have colons
128                            //so remove the colon from the timezone offset
129                            dateStr = dateStr.replaceAll("([-\\+]\\d{2}):(\\d{2})$", "$1$2");
130                            break;
131                    case UTC_TIME_BASIC:
132                    case UTC_TIME_EXTENDED:
133                            //SimpleDateFormat doesn't recognize "Z"
134                            dateStr = dateStr.replace("Z", "+0000");
135                            break;
136                    }
137    
138                    //parse the date
139                    DateFormat df = format.getParseDateFormat();
140                    if (timezone != null) {
141                            df.setTimeZone(timezone);
142                    }
143                    try {
144                            return df.parse(dateStr);
145                    } catch (ParseException e) {
146                            //should never be thrown because the string is checked against a regex
147                            throw new IllegalArgumentException("Date string is not in a valid ISO-8601 format.");
148                    }
149            }
150    
151            /**
152             * Determines whether a date string has a time component.
153             * @param dateStr the date string (e.g. "20130601T120000")
154             * @return true if it has a time component, false if not
155             */
156            public static boolean dateHasTime(String dateStr) {
157                    return dateStr.contains("T");
158            }
159    
160            /**
161             * Determines whether a date string is in UTC time or has a timezone offset.
162             * @param dateStr the date string (e.g. "20130601T120000Z",
163             * "20130601T120000-0400")
164             * @return true if it has a timezone, false if not
165             */
166            public static boolean dateHasTimezone(String dateStr) {
167                    return dateStr.endsWith("Z") || dateStr.matches(".*?[-+]\\d\\d:?\\d\\d");
168            }
169    
170            /**
171             * Parses a timezone that's in ISO8601 format.
172             * @param offsetStr the timezone offset string (e.g. "-0500" or "-05:00")
173             * @return the hour offset (index 0) and the minute offset (index 1)
174             * @throws IllegalArgumentException if the timezone string isn't in the
175             * right format
176             */
177            public static int[] parseTimeZone(String offsetStr) {
178                    Matcher m = timeZoneRegex.matcher(offsetStr);
179    
180                    if (!m.find()) {
181                            throw new IllegalArgumentException("Offset string is not in ISO8610 format: " + offsetStr);
182                    }
183    
184                    String sign = m.group(1);
185                    boolean positive;
186                    if ("-".equals(sign)) {
187                            positive = false;
188                    } else {
189                            positive = true;
190                    }
191    
192                    String hourStr = m.group(2);
193                    int hourOffset = Integer.parseInt(hourStr);
194                    if (!positive) {
195                            hourOffset *= -1;
196                    }
197    
198                    String minuteStr = m.group(4);
199                    int minuteOffset = (minuteStr == null) ? 0 : Integer.parseInt(minuteStr);
200    
201                    return new int[] { hourOffset, minuteOffset };
202            }
203    
204            /**
205             * Formats a {@link TimeZone} object according to ISO8601 rules.
206             * 
207             * @param timeZone the timezone to format
208             * @param extended true to use "extended" format, false not to. Extended
209             * format will put a colon between the hour and minute.
210             * @return the formatted timezone (e.g. "+0530" or "+05:30")
211             */
212            public static String formatTimeZone(TimeZone timeZone, boolean extended) {
213                    int hours = timeZone.getRawOffset() / 1000 / 60 / 60;
214                    int minutes = Math.abs((timeZone.getRawOffset() / 1000) / 60) % 60;
215                    return formatTimeZone(hours, minutes, extended);
216            }
217    
218            /**
219             * Formats a timezone offset according to ISO8601 rules.
220             * 
221             * @param hourOffset the hour offset
222             * @param minuteOffset the minute offset (between 0 and 59)
223             * @param extended true to use "extended" format, false not to. Extended
224             * format will put a colon between the hour and minute.
225             * @return the formatted timezone (e.g. "+0530" or "+05:30")
226             */
227            public static String formatTimeZone(int hourOffset, int minuteOffset, boolean extended) {
228                    StringBuilder sb = new StringBuilder();
229                    boolean positive = hourOffset >= 0;
230    
231                    sb.append(positive ? '+' : '-');
232    
233                    hourOffset = Math.abs(hourOffset);
234                    if (hourOffset < 10) {
235                            sb.append('0');
236                    }
237                    sb.append(hourOffset);
238    
239                    if (extended) {
240                            sb.append(':');
241                    }
242    
243                    if (minuteOffset < 10) {
244                            sb.append('0');
245                    }
246                    sb.append(minuteOffset);
247    
248                    return sb.toString();
249            }
250    
251            /**
252             * Gets the {@link TimeZone} object that corresponds to the given ID.
253             * @param timezoneId the timezone ID (e.g. "America/New_York")
254             * @return the timezone object or null if not found
255             */
256            public static TimeZone parseTimeZoneId(String timezoneId) {
257                    TimeZone timezone = TimeZone.getTimeZone(timezoneId);
258                    return "GMT".equals(timezone.getID()) ? null : timezone;
259            }
260    
261            private ICalDateFormatter() {
262                    //hide constructor
263            }
264    }
