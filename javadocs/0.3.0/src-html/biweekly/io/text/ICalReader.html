058    /**
059     * <p>
060     * Parses {@link ICalendar} objects from an iCalendar data stream.
061     * </p>
062     * <p>
063     * <b>Example:</b>
064     * 
065     * <pre>
066     * Reader reader = ...
067     * ICalReader icalReader = new ICalReader(reader);
068     * ICalendar ical;
069     * while ((ical = icalReader.readNext()) != null){
070     *   ...
071     * }
072     * icalReader.close();
073     * </pre>
074     * 
075     * </p>
076     * @author Michael Angstadt
077     * @see <a href="http://tools.ietf.org/html/rfc5545">RFC 5545</a>
078     */
079    public class ICalReader implements Closeable {
080            private static final ICalendarMarshaller icalMarshaller = (ICalendarMarshaller) ComponentLibrary.getMarshaller(ICalendar.class);
081            private final List<String> warnings = new ArrayList<String>();
082            private final Map<String, ICalPropertyMarshaller<? extends ICalProperty>> propertyMarshallers = new HashMap<String, ICalPropertyMarshaller<? extends ICalProperty>>(0);
083            private final Map<String, ICalComponentMarshaller<? extends ICalComponent>> componentMarshallers = new HashMap<String, ICalComponentMarshaller<? extends ICalComponent>>(0);
084            private final ICalRawReader reader;
085    
086            /**
087             * Creates a reader that parses iCalendar objects from a string.
088             * @param string the string
089             */
090            public ICalReader(String string) {
091                    this(new StringReader(string));
092            }
093    
094            /**
095             * Creates a reader that parses iCalendar objects from an input stream.
096             * @param in the input stream
097             */
098            public ICalReader(InputStream in) {
099                    this(new InputStreamReader(in));
100            }
101    
102            /**
103             * Creates a reader that parses iCalendar objects from a file.
104             * @param file the file
105             * @throws FileNotFoundException if the file doesn't exist
106             */
107            public ICalReader(File file) throws FileNotFoundException {
108                    this(new FileReader(file));
109            }

111            /**
112             * Creates a reader that parses iCalendar objects from a reader.
113             * @param reader the reader
114             */
115            public ICalReader(Reader reader) {
116                    this.reader = new ICalRawReader(reader);
117            }

119            /**
120             * Gets whether the reader will decode parameter values that use circumflex
121             * accent encoding (enabled by default). This escaping mechanism allows
122             * newlines and double quotes to be included in parameter values.
123             * @return true if circumflex accent decoding is enabled, false if not
124             * @see ICalRawReader#isCaretDecodingEnabled()
125             */
126            public boolean isCaretDecodingEnabled() {
127                    return reader.isCaretDecodingEnabled();
128            }

130            /**
131             * Sets whether the reader will decode parameter values that use circumflex
132             * accent encoding (enabled by default). This escaping mechanism allows
133             * newlines and double quotes to be included in parameter values.
134             * @param enable true to use circumflex accent decoding, false not to
135             * @see ICalRawReader#setCaretDecodingEnabled(boolean)
136             */
137            public void setCaretDecodingEnabled(boolean enable) {
138                    reader.setCaretDecodingEnabled(enable);
139            }

141            /**
142             * Registers a marshaller for an experimental property.
143             * @param marshaller the marshaller to register
144             */
145            public void registerMarshaller(ICalPropertyMarshaller<? extends ICalProperty> marshaller) {
146                    propertyMarshallers.put(marshaller.getPropertyName().toUpperCase(), marshaller);
147            }
148    
149            /**
150             * Registers a marshaller for an experimental component.
151             * @param marshaller the marshaller to register
152             */
153            public void registerMarshaller(ICalComponentMarshaller<? extends ICalComponent> marshaller) {
154                    componentMarshallers.put(marshaller.getComponentName().toUpperCase(), marshaller);
155            }
156    
157            /**
158             * Gets the warnings from the last iCalendar object that was unmarshalled.
159             * This list is reset every time a new iCalendar object is read.
160             * @return the warnings or empty list if there were no warnings
161             */
162            public List<String> getWarnings() {
163                    return new ArrayList<String>(warnings);
164            }
165    
166            /**
167             * Reads the next iCalendar object.
168             * @return the next iCalendar object or null if there are no more
169             * @throws IOException if there's a problem reading from the stream
170             */
171            public ICalendar readNext() throws IOException {
172                    if (reader.eof()) {
173                            return null;
174                    }
175    
176                    warnings.clear();
177    
178                    ICalDataStreamListenerImpl listener = new ICalDataStreamListenerImpl();
179                    reader.start(listener);
180    
181                    if (!listener.dataWasRead) {
182                            //EOF was reached without reading anything
183                            return null;
184                    }
185    
186                    ICalendar ical;
187                    if (listener.orphanedComponents.isEmpty()) {
188                            //there were no components in the iCalendar object
189                            ical = icalMarshaller.emptyInstance();
190                    } else {
191                            ICalComponent first = listener.orphanedComponents.get(0);
192                            if (first instanceof ICalendar) {
193                                    //this is the code-path that valid iCalendar objects should reach
194                                    ical = (ICalendar) first;
195                            } else {
196                                    ical = icalMarshaller.emptyInstance();
197                                    for (ICalComponent component : listener.orphanedComponents) {
198                                            ical.addComponent(component);
199                                    }
200                            }
201                    }
202    
203                    //add any properties that were not part of a component (will never happen if the iCalendar object is valid)
204                    for (ICalProperty property : listener.orphanedProperties) {
205                            ical.addProperty(property);
206                    }
207    
208                    return ical;
209            }
210    
211            /**
212             * Finds a component marshaller.
213             * @param componentName the name of the component
214             * @return the component marshallerd
215             */
216            private ICalComponentMarshaller<? extends ICalComponent> findComponentMarshaller(String componentName) {
217                    ICalComponentMarshaller<? extends ICalComponent> m = componentMarshallers.get(componentName.toUpperCase());
218                    if (m == null) {
219                            m = ComponentLibrary.getMarshaller(componentName);
220                            if (m == null) {
221                                    m = new RawComponentMarshaller(componentName);
222                            }
223                    }
224                    return m;
225            }
226    
227            /**
228             * Finds a property marshaller.
229             * @param propertyName the name of the property
230             * @return the property marshaller
231             */
232            private ICalPropertyMarshaller<? extends ICalProperty> findPropertyMarshaller(String propertyName) {
233                    ICalPropertyMarshaller<? extends ICalProperty> m = propertyMarshallers.get(propertyName.toUpperCase());
234                    if (m == null) {
235                            m = PropertyLibrary.getMarshaller(propertyName);
236                            if (m == null) {
237                                    m = new RawPropertyMarshaller(propertyName);
238                            }
239                    }
240                    return m;
241            }

243            //TODO how to unmarshal the alarm components (a different class should be created, depending on the ACTION property)
244            //TODO buffer properties in a list before the component class is created
245            private class ICalDataStreamListenerImpl implements ICalRawReader.ICalDataStreamListener {
246                    private final String icalComponentName = ComponentLibrary.getMarshaller(ICalendar.class).getComponentName();
247    
248                    private List<ICalProperty> orphanedProperties = new ArrayList<ICalProperty>();
249                    private List<ICalComponent> orphanedComponents = new ArrayList<ICalComponent>();
250    
251                    private List<ICalComponent> componentStack = new ArrayList<ICalComponent>();
252                    private List<String> componentNamesStack = new ArrayList<String>();
253                    private boolean dataWasRead = false;
254    
255                    public void beginComponent(String name) {
256                            dataWasRead = true;
257    
258                            ICalComponent parentComponent = getCurrentComponent();
259    
260                            ICalComponentMarshaller<? extends ICalComponent> m = findComponentMarshaller(name);
261                            ICalComponent component = m.emptyInstance();
262                            componentStack.add(component);
263                            componentNamesStack.add(name);
264    
265                            if (parentComponent == null) {
266                                    orphanedComponents.add(component);
267                            } else {
268                                    parentComponent.addComponent(component);
269                            }
270                    }
271    
272                    public void readProperty(String name, ICalParameters parameters, String value) {
273                            dataWasRead = true;
274    
275                            ICalPropertyMarshaller<? extends ICalProperty> m = findPropertyMarshaller(name);
276                            ICalProperty property = null;
277                            try {
278                                    Result<? extends ICalProperty> result = m.parseText(value, parameters);
279    
280                                    for (String warning : result.getWarnings()) {
281                                            addWarning(warning, name);
282                                    }
283    
284                                    property = result.getValue();
285                            } catch (SkipMeException e) {
286                                    if (e.getMessage() == null) {
287                                            addWarning("Property has requested that it be skipped.", name);
288                                    } else {
289                                            addWarning("Property has requested that it be skipped: " + e.getMessage(), name);
290                                    }
291                            } catch (CannotParseException e) {
292                                    if (e.getMessage() == null) {
293                                            addWarning("Property value could not be unmarshalled: " + value, name);
294                                    } else {
295                                            addWarning("Property value could not be unmarshalled.\n  Value: " + value + "\n  Reason: " + e.getMessage(), name);
296                                    }
297                                    property = new RawProperty(name, value);
298                            }
299    
300                            if (property != null) {
301                                    ICalComponent parentComponent = getCurrentComponent();
302                                    if (parentComponent == null) {
303                                            orphanedProperties.add(property);
304                                    } else {
305                                            parentComponent.addProperty(property);
306                                    }
307                            }
308                    }
309    
310                    public void endComponent(String name) {
311                            //stop reading when "END:VCALENDAR" is reached
312                            if (icalComponentName.equalsIgnoreCase(name)) {
313                                    throw new ICalRawReader.StopReadingException();
314                            }
315    
316                            //find the component that this END property matches up with
317                            int popIndex = -1;
318                            for (int i = componentStack.size() - 1; i >= 0; i--) {
319                                    String n = componentNamesStack.get(i);
320                                    if (n.equalsIgnoreCase(name)) {
321                                            popIndex = i;
322                                            break;
323                                    }
324                            }
325                            if (popIndex == -1) {
326                                    //END property does not match up with any BEGIN properties, so ignore
327                                    addWarning("Ignoring END property that does not match up with any BEGIN properties: " + name, "END");
328                                    return;
329                            }
330    
331                            componentStack = componentStack.subList(0, popIndex);
332                            componentNamesStack = componentNamesStack.subList(0, popIndex);
333                    }
334    
335                    public void invalidLine(String line) {
336                            addWarning("Skipping malformed line: \"" + line + "\"");
337                    }
338    
339                    public void valuelessParameter(String propertyName, String parameterName) {
340                            addWarning("Value-less parameter encountered: " + parameterName, propertyName);
341                    }
342    
343                    private ICalComponent getCurrentComponent() {
344                            if (componentStack.isEmpty()) {
345                                    return null;
346                            }
347                            return componentStack.get(componentStack.size() - 1);
348                    }
349            }
350    
351            private void addWarning(String message) {
352                    addWarning(message, null);
353            }
354    
355            private void addWarning(String message, String propertyName) {
356                    addWarning(message, propertyName, reader.getLineNum());
357            }
358    
359            private void addWarning(String message, String propertyName, int lineNum) {
360                    StringBuilder sb = new StringBuilder();
361                    sb.append("Line ").append(lineNum);
362                    if (propertyName != null) {
363                            sb.append(" (").append(propertyName).append(" property)");
364                    }
365                    sb.append(": ").append(message);
366    
367                    warnings.add(sb.toString());
368            }

372             */
373            //@Override

375                    reader.close();
376            }
