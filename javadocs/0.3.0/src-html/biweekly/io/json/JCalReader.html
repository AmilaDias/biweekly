061    /**
062     * <p>
063     * Parses {@link ICalendar} objects from a jCal data stream (JSON).
064     * </p>
065     * <p>
066     * <b>Example:</b>
067     * 
068     * <pre>
069     * Reader reader = ...
070     * JCalReader jcalReader = new JCalReader(reader);
071     * ICalendar ical;
072     * while ((ical = jcalReader.readNext()) != null){
073     *   ...
074     * }
075     * jcalReader.close();
076     * </pre>
077     * 
078     * </p>
079     * @author Michael Angstadt
080     * @see <a href="http://tools.ietf.org/html/draft-ietf-jcardcal-jcal-05">jCal
081     * draft</a>
082     */
083    public class JCalReader implements Closeable {
084            private static final ICalendarMarshaller icalMarshaller = (ICalendarMarshaller) ComponentLibrary.getMarshaller(ICalendar.class);
085            private final JCalRawReader reader;
086            private final List<String> warnings = new ArrayList<String>();
087            private final Map<String, ICalPropertyMarshaller<? extends ICalProperty>> propertyMarshallers = new HashMap<String, ICalPropertyMarshaller<? extends ICalProperty>>(0);
088            private final Map<String, ICalComponentMarshaller<? extends ICalComponent>> componentMarshallers = new HashMap<String, ICalComponentMarshaller<? extends ICalComponent>>(0);
089    
090            /**
091             * Creates a jCard reader.
092             * @param json the JSON string
093             */
094            public JCalReader(String json) {
095                    this(new StringReader(json));
096            }
097    
098            /**
099             * Creates a jCard reader.
100             * @param in the input stream to read the vCards from
101             */
102            public JCalReader(InputStream in) {
103                    this(new InputStreamReader(in));
104            }
105    
106            /**
107             * Creates a jCard reader.
108             * @param file the file to read the vCards from
109             * @throws FileNotFoundException if the file doesn't exist
110             */
111            public JCalReader(File file) throws FileNotFoundException {
112                    this(new FileReader(file));
113            }
114    
115            /**
116             * Creates a jCard reader.
117             * @param reader the reader to read the vCards from
118             */
119            public JCalReader(Reader reader) {
120                    this.reader = new JCalRawReader(reader);
121            }
122    
123            /**
124             * Gets the warnings from the last iCalendar object that was unmarshalled.
125             * This list is reset every time a new iCalendar object is read.
126             * @return the warnings or empty list if there were no warnings
127             */
128            public List<String> getWarnings() {
129                    return new ArrayList<String>(warnings);
130            }
131    
132            /**
133             * Registers a marshaller for an experimental property.
134             * @param marshaller the marshaller to register
135             */
136            public void registerMarshaller(ICalPropertyMarshaller<? extends ICalProperty> marshaller) {
137                    propertyMarshallers.put(marshaller.getPropertyName().toLowerCase(), marshaller);
138            }
139    
140            /**
141             * Registers a marshaller for an experimental component.
142             * @param marshaller the marshaller to register
143             */
144            public void registerMarshaller(ICalComponentMarshaller<? extends ICalComponent> marshaller) {
145                    componentMarshallers.put(marshaller.getComponentName().toLowerCase(), marshaller);
146            }
147    
148            /**
149             * Reads the next iCalendar object from the JSON data stream.
150             * @return the iCalendar object or null if there are no more
151             * @throws JCalParseException if the jCal syntax is incorrect (the JSON
152             * syntax may be valid, but it is not in the correct jCal format).
153             * @throws JsonParseException if the JSON syntax is incorrect
154             * @throws IOException if there is a problem reading from the data stream
155             */
156            public ICalendar readNext() throws IOException {
157                    if (reader.eof()) {
158                            return null;
159                    }
160    
161                    warnings.clear();
162    
163                    JCalDataStreamListenerImpl listener = new JCalDataStreamListenerImpl();
164                    reader.readNext(listener);
165                    return listener.getICalendar();
166            }
167    
168            /**
169             * Finds a component marshaller.
170             * @param componentName the name of the component
171             * @return the component marshallerd
172             */
173            private ICalComponentMarshaller<? extends ICalComponent> findComponentMarshaller(String componentName) {
174                    ICalComponentMarshaller<? extends ICalComponent> m = componentMarshallers.get(componentName.toLowerCase());
175                    if (m == null) {
176                            m = ComponentLibrary.getMarshaller(componentName);
177                            if (m == null) {
178                                    m = new RawComponentMarshaller(componentName);
179                            }
180                    }
181                    return m;
182            }
183    
184            /**
185             * Finds a property marshaller.
186             * @param propertyName the name of the property
187             * @return the property marshaller
188             */
189            private ICalPropertyMarshaller<? extends ICalProperty> findPropertyMarshaller(String propertyName) {
190                    ICalPropertyMarshaller<? extends ICalProperty> m = propertyMarshallers.get(propertyName.toLowerCase());
191                    if (m == null) {
192                            m = PropertyLibrary.getMarshaller(propertyName);
193                            if (m == null) {
194                                    m = new RawPropertyMarshaller(propertyName);
195                            }
196                    }
197                    return m;
198            }
199    
200            private void addWarning(String message, String propertyName) {
201                    StringBuilder sb = new StringBuilder();
202                    sb.append("Line ").append(reader.getLineNum());
203                    if (propertyName != null) {
204                            sb.append(" (").append(propertyName).append(" property)");
205                    }
206                    sb.append(": ").append(message);
207    
208                    warnings.add(sb.toString());
209            }
210    
211            //@Override
212            public void close() throws IOException {
213                    reader.close();
214            }
215    
216            private class JCalDataStreamListenerImpl implements JCalDataStreamListener {
217                    private final Map<List<String>, ICalComponent> components = new HashMap<List<String>, ICalComponent>();
218    
219                    public void readProperty(List<String> componentHierarchy, String propertyName, ICalParameters parameters, JCalValue value) {
220                            //get the component that the property belongs to
221                            ICalComponent parent = components.get(componentHierarchy);
222    
223                            //unmarshal the property
224                            ICalPropertyMarshaller<? extends ICalProperty> m = findPropertyMarshaller(propertyName);
225                            ICalProperty property = null;
226                            try {
227                                    Result<? extends ICalProperty> result = m.parseJson(value, parameters);
228    
229                                    for (String warning : result.getWarnings()) {
230                                            addWarning(warning, propertyName);
231                                    }
232    
233                                    property = result.getValue();
234                            } catch (SkipMeException e) {
235                                    if (e.getMessage() == null) {
236                                            addWarning("Property has requested that it be skipped.", propertyName);
237                                    } else {
238                                            addWarning("Property has requested that it be skipped: " + e.getMessage(), propertyName);
239                                    }
240                            } catch (CannotParseException e) {
241                                    if (e.getMessage() == null) {
242                                            addWarning("Property value could not be unmarshalled: " + value, propertyName);
243                                    } else {
244                                            addWarning("Property value could not be unmarshalled.\n  Value: " + value + "\n  Reason: " + e.getMessage(), propertyName);
245                                    }
246    
247                                    Result<? extends ICalProperty> result = new RawPropertyMarshaller(propertyName).parseJson(value, parameters);
248                                    for (String warning : result.getWarnings()) {
249                                            addWarning(warning, propertyName);
250                                    }
251                                    property = result.getValue();
252                            }
253    
254                            if (property != null) {
255                                    parent.addProperty(property);
256                            }
257                    }
258    
259                    public void readComponent(List<String> parentHierarchy, String componentName) {
260                            ICalComponentMarshaller<? extends ICalComponent> m = findComponentMarshaller(componentName);
261                            ICalComponent component = m.emptyInstance();
262    
263                            ICalComponent parent = components.get(parentHierarchy);
264                            if (parent != null) {
265                                    parent.addComponent(component);
266                            }
267    
268                            List<String> hierarchy = new ArrayList<String>(parentHierarchy);
269                            hierarchy.add(componentName);
270                            components.put(hierarchy, component);
271                    }
272    
273                    public ICalendar getICalendar() {
274                            if (components.isEmpty()) {
275                                    //EOF
276                                    return null;
277                            }
278    
279                            ICalComponent component = components.get(Arrays.asList(icalMarshaller.getComponentName().toLowerCase()));
280                            if (component == null) {
281                                    //should never happen because the parser always looks for a "vcalendar" component
282                                    return null;
283                            }
284    
285                            if (component instanceof ICalendar) {
286                                    //should happen every time
287                                    return (ICalendar) component;
288                            }
289    
290                            //this will only happen if the user decides to override the ICalendarMarshaller for some reason
291                            ICalendar ical = icalMarshaller.emptyInstance();
292                            ical.addComponent(component);
293                            return ical;
294                    }
295            }
