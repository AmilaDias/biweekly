040    /**
041     * Marshals {@link RequestStatus} properties.
042     * @author Michael Angstadt
043     */
044    public class RequestStatusMarshaller extends ICalPropertyMarshaller<RequestStatus> {
045            public RequestStatusMarshaller() {
046                    super(RequestStatus.class, "REQUEST-STATUS");
047            }
048    
049            @Override
050            protected String _writeText(RequestStatus property) {
051                    List<String> components = new ArrayList<String>();
052                    addComponent(property.getExceptionText(), components);
053                    addComponent(property.getDescription(), components);
054                    addComponent(property.getStatusCode(), components);
055                    Collections.reverse(components);
056    
057                    return StringUtils.join(components, ";", new JoinCallback<String>() {
058                            public void handle(StringBuilder sb, String component) {
059                                    sb.append(escape(component));
060                            }
061                    });
062            }
063    
064            private void addComponent(String component, List<String> components) {
065                    if (component != null) {
066                            components.add(component);
067                    } else if (!components.isEmpty()) {
068                            components.add("");
069                    }
070            }
071    
072            @Override
073            protected RequestStatus _parseText(String value, ICalParameters parameters, List<String> warnings) {
074                    List<String> split = split(value, ";").unescape(true).split();
075                    return parse(split);
076            }
077    
078            @Override
079            protected void _writeXml(RequestStatus property, XCalElement element) {
080                    String code = property.getStatusCode();
081                    if (code != null) {
082                            element.append("code", code);
083                    }
084    
085                    String description = property.getDescription();
086                    if (description != null) {
087                            element.append("description", description);
088                    }
089    
090                    String data = property.getExceptionText();
091                    if (data != null) {
092                            element.append("data", data);
093                    }
094            }
095    
096            @Override
097            protected RequestStatus _parseXml(XCalElement element, ICalParameters parameters, List<String> warnings) {
098                    RequestStatus requestStatus = new RequestStatus(element.first("code"));
099                    requestStatus.setDescription(element.first("description"));
100                    requestStatus.setExceptionText(element.first("data"));
101                    return requestStatus;
102            }
103    
104            @Override
105            protected JCalValue _writeJson(RequestStatus property) {
106                    List<String> components = new ArrayList<String>();
107                    addComponent(property.getExceptionText(), components);
108                    addComponent(property.getDescription(), components);
109                    addComponent(property.getStatusCode(), components);
110                    Collections.reverse(components);
111    
112                    return JCalValue.structured(Value.TEXT, components);
113            }
114    
115            @Override
116            protected RequestStatus _parseJson(JCalValue value, ICalParameters parameters, List<String> warnings) {
117                    List<String> values = value.getStructured();
118                    return parse(values);
119            }
120    
121            private RequestStatus parse(List<String> values) {
122                    RequestStatus requestStatus = new RequestStatus(values.isEmpty() ? null : values.get(0));
123    
124                    if (values.size() > 1) {
125                            requestStatus.setDescription(values.get(1));
126                    }
127                    if (values.size() > 2) {
128                            requestStatus.setExceptionText(values.get(2));
129                    }
130    
131                    return requestStatus;
132            }
133    }
