412            /**
413             * Adds a single time period for which the person is free or busy (for
414             * example, "free" between 1pm-3pm). This method will look for an existing
415             * property that has the given {@link FreeBusyType} and add the time period
416             * to it, or create a new property is one cannot be found.
417             * @param type the availability type (e.g. "free" or "busy")
418             * @param start the start date-time
419             * @param end the end date-time
420             * @return the property that was created/modified
421             * @see <a href="http://tools.ietf.org/html/rfc5545#page-100">RFC 5545
422             * p.100-1</a>
423             */
424            public FreeBusy addFreeBusy(FreeBusyType type, Date start, Date end) {
425                    FreeBusy found = findByFbType(type);
426                    found.addValue(start, end);
427                    return found;
428            }
429    
430            /**
431             * Adds a single time period for which the person is free or busy (for
432             * example, "free" for 2 hours after 1pm). This method will look for an
433             * existing property that has the given {@link FreeBusyType} and add the
434             * time period to it, or create a new property is one cannot be found.
435             * @param type the availability type (e.g. "free" or "busy")
436             * @param start the start date-time
437             * @param duration the length of time
438             * @return the property that was created/modified
439             * @see <a href="http://tools.ietf.org/html/rfc5545#page-100">RFC 5545
440             * p.100-1</a>
441             */
442            public FreeBusy addFreeBusy(FreeBusyType type, Date start, Duration duration) {
443                    FreeBusy found = findByFbType(type);
444                    found.addValue(start, duration);
445                    return found;
446            }
447    
448            private FreeBusy findByFbType(FreeBusyType type) {
449                    FreeBusy found = null;
450    
451                    for (FreeBusy fb : getFreeBusy()) {
452                            if (fb.getType() == type) {
453                                    found = fb;
454                                    break;
455                            }
456                    }
457    
458                    if (found == null) {
459                            found = new FreeBusy();
460                            found.setType(type);
461                            addFreeBusy(found);
462                    }
463                    return found;
464            }

486            @Override
487            protected void validate(List<ICalComponent> components, List<String> warnings) {
488                    if (getUid() == null) {
489                            warnings.add(Uid.class.getSimpleName() + " is not set (it is a required property).");
490                    }
491                    if (getDateTimeStamp() == null) {
492                            warnings.add(DateTimeStamp.class.getSimpleName() + " is not set (it is a required property).");
493                    }
494    
495                    DateStart dateStart = getDateStart();
496                    DateEnd dateEnd = getDateEnd();
497    
498                    if (dateEnd != null && dateStart == null) {
499                            warnings.add("A " + DateStart.class.getSimpleName() + " property must be defined if a " + DateEnd.class.getSimpleName() + " property is defined.");
500                    }
501    
502                    if (dateStart != null && dateStart.getValue() != null && !dateStart.hasTime()) {
503                            warnings.add(DateStart.class.getSimpleName() + " properties in free/busy components must always have a time component.");
504                    }
505    
506                    if (dateEnd != null && dateEnd.getValue() != null && !dateEnd.hasTime()) {
507                            warnings.add(DateEnd.class.getSimpleName() + " properties in free/busy components must always have a time component.");
508                    }
509    
510                    if (dateStart != null && dateEnd != null) {
511                            Date start = dateStart.getValue();
512                            Date end = dateEnd.getValue();
513                            if (start != null && end != null && start.compareTo(end) >= 0) {
514                                    warnings.add(DateStart.class.getSimpleName() + " must come before " + DateEnd.class.getSimpleName() + ".");
515                            }
516                    }
517            }
