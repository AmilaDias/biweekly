057    /**
058     * <p>
059     * Parses {@link ICalendar} objects from an iCalendar data stream.
060     * </p>
061     * 
062     * <pre>
063     * Reader reader = ...
064     * ICalReader icalReader = new ICalReader(reader);
065     * ICalendar ical;
066     * while ((ical = icalReader.readNext()) != null){
067     *   ...
068     * }
069     * icalReader.close();
070     * </pre>
071     * @author Michael Angstadt
072     */
073    public class ICalReader implements IParser {
074            private final List<String> warnings = new ArrayList<String>();
075            private final Map<String, ICalPropertyMarshaller<? extends ICalProperty>> propertyMarshallers = new HashMap<String, ICalPropertyMarshaller<? extends ICalProperty>>(0);
076            private final Map<String, ICalComponentMarshaller<? extends ICalComponent>> componentMarshallers = new HashMap<String, ICalComponentMarshaller<? extends ICalComponent>>(0);
077            private final ICalRawReader reader;
078    
079            /**
080             * @param str the string to read the iCals from
081             */
082            public ICalReader(String str) {
083                    this(new StringReader(str));
084            }
085    
086            /**
087             * @param in the input stream to read the iCals from
088             */
089            public ICalReader(InputStream in) {
090                    this(new InputStreamReader(in));
091            }
092    
093            /**
094             * @param file the file to read the iCals from
095             * @throws FileNotFoundException if the file doesn't exist
096             */
097            public ICalReader(File file) throws FileNotFoundException {
098                    this(new FileReader(file));
099            }
100    
101            /**
102             * @param reader the reader to read the iCal from
103             */
104            public ICalReader(Reader reader) {
105                    this.reader = new ICalRawReader(reader);
106            }
107    
108            /**
109             * Gets whether the reader will decode parameter values that use circumflex
110             * accent encoding. This escaping mechanism allows newlines and double
111             * quotes to be included in parameter values. It is enabled by default.
112             * @return true if circumflex accent decoding is enabled, false if not
113             * @see ICalRawReader#isCaretDecodingEnabled()
114             */
115            public boolean isCaretDecodingEnabled() {
116                    return reader.isCaretDecodingEnabled();
117            }
118    
119            /**
120             * Sets whether the reader will decode parameter values that use circumflex
121             * accent encoding. This escaping mechanism allows newlines and double
122             * quotes to be included in parameter values. It is enabled by default.
123             * @param enable true to use circumflex accent decoding, false not to
124             * @see ICalRawReader#setCaretDecodingEnabled(boolean)
125             */
126            public void setCaretDecodingEnabled(boolean enable) {
127                    reader.setCaretDecodingEnabled(enable);
128            }
129    
130            //@Override
131            public void registerMarshaller(ICalPropertyMarshaller<? extends ICalProperty> marshaller) {
132                    propertyMarshallers.put(marshaller.getPropertyName().toUpperCase(), marshaller);
133            }
134    
135            //@Override
136            public void registerMarshaller(ICalComponentMarshaller<? extends ICalComponent> marshaller) {
137                    componentMarshallers.put(marshaller.getComponentName().toUpperCase(), marshaller);
138            }
139    
140            //@Override
141            public List<String> getWarnings() {
142                    return new ArrayList<String>(warnings);
143            }
144    
145            //@Override
146            public ICalendar readNext() throws IOException {
147                    if (reader.eof()) {
148                            return null;
149                    }
150    
151                    warnings.clear();
152    
153                    ICalDataStreamListenerImpl listener = new ICalDataStreamListenerImpl();
154                    reader.start(listener);
155    
156                    if (!listener.dataWasRead) {
157                            //EOF was reached without reading anything
158                            return null;
159                    }
160    
161                    ICalendar ical;
162                    if (listener.orphanedComponents.isEmpty()) {
163                            //there were no components in the iCalendar object
164                            ical = new ICalendar();
165                            ical.getProperties().clear(); //clear properties that are created in the constructor
166                    } else {
167                            ICalComponent first = listener.orphanedComponents.get(0);
168                            if (first instanceof ICalendar) {
169                                    //this is the code-path that valid iCalendar objects should reach
170                                    ical = (ICalendar) first;
171                            } else {
172                                    ical = new ICalendar();
173                                    ical.getProperties().clear(); //clear properties that are created in the constructor
174                                    for (ICalComponent component : listener.orphanedComponents) {
175                                            ical.addComponent(component);
176                                    }
177                            }
178                    }
179    
180                    //add any properties that were not part of a component (will never happen if the iCalendar object is valid)
181                    for (ICalProperty property : listener.orphanedProperties) {
182                            ical.addProperty(property);
183                    }
184    
185                    return ical;
186            }
187    
188            /**
189             * Finds a component marshaller.
190             * @param componentName the name of the component
191             * @return the component marshallerd
192             */
193            private ICalComponentMarshaller<? extends ICalComponent> findComponentMarshaller(final String componentName) {
194                    ICalComponentMarshaller<? extends ICalComponent> m = componentMarshallers.get(componentName.toUpperCase());
195                    if (m == null) {
196                            m = ComponentLibrary.getMarshaller(componentName);
197                            if (m == null) {
198                                    m = new RawComponentMarshaller(componentName);
199                            }
200                    }
201                    return m;
202            }
203    
204            /**
205             * Finds a property marshaller.
206             * @param propertyName the name of the property
207             * @return the property marshaller
208             */
209            private ICalPropertyMarshaller<? extends ICalProperty> findPropertyMarshaller(String propertyName) {
210                    ICalPropertyMarshaller<? extends ICalProperty> m = propertyMarshallers.get(propertyName);
211                    if (m == null) {
212                            m = PropertyLibrary.getMarshaller(propertyName);
213                            if (m == null) {
214                                    m = new RawPropertyMarshaller(propertyName);
215                            }
216                    }
217                    return m;
218            }
219    
220            //TODO how to unmarshal the alarm components (a different class should be created, depending on the ACTION property)
221            //TODO buffer properties in a list before the component class is created
222            private class ICalDataStreamListenerImpl implements ICalRawReader.ICalDataStreamListener {
223                    private final String icalComponentName = ComponentLibrary.getMarshaller(ICalendar.class).getComponentName();
224    
225                    private List<ICalProperty> orphanedProperties = new ArrayList<ICalProperty>();
226                    private List<ICalComponent> orphanedComponents = new ArrayList<ICalComponent>();
227    
228                    private List<ICalComponent> componentStack = new ArrayList<ICalComponent>();
229                    private List<String> componentNamesStack = new ArrayList<String>();
230                    private boolean dataWasRead = false;
231    
232                    public void beginComponent(String name) {
233                            dataWasRead = true;
234    
235                            ICalComponent parentComponent = getCurrentComponent();
236    
237                            ICalComponentMarshaller<? extends ICalComponent> m = findComponentMarshaller(name);
238                            ICalComponent component = m.newInstance();
239                            component.getProperties().clear(); //clear properties that were created in the constructor
240                            componentStack.add(component);
241                            componentNamesStack.add(name);
242    
243                            if (parentComponent == null) {
244                                    orphanedComponents.add(component);
245                            } else {
246                                    parentComponent.addComponent(component);
247                            }
248                    }
249    
250                    public void readProperty(String name, ICalParameters parameters, String value) {
251                            dataWasRead = true;
252    
253                            ICalPropertyMarshaller<? extends ICalProperty> m = findPropertyMarshaller(name);
254                            ICalProperty property = null;
255                            try {
256                                    Result<? extends ICalProperty> result = m.parseText(value, parameters);
257    
258                                    for (String warning : result.getWarnings()) {
259                                            addWarning(warning, name);
260                                    }
261    
262                                    property = result.getValue();
263                            } catch (SkipMeException e) {
264                                    addWarning("Property has requested that it be skipped: " + e.getMessage(), name);
265                            } catch (CannotParseException e) {
266                                    if (e.getMessage() == null) {
267                                            addWarning("Property value could not be unmarshalled: " + value, name);
268                                    } else {
269                                            addWarning("Property value could not be unmarshalled.\n  Value: " + value + "\n  Reason: " + e.getMessage(), name);
270                                    }
271                                    property = new RawProperty(name, value);
272                            }
273    
274                            if (property != null) {
275                                    ICalComponent parentComponent = getCurrentComponent();
276                                    if (parentComponent == null) {
277                                            orphanedProperties.add(property);
278                                    } else {
279                                            parentComponent.addProperty(property);
280                                    }
281                            }
282                    }
283    
284                    public void endComponent(String name) {
285                            //stop reading when "END:VCALENDAR" is reached
286                            if (icalComponentName.equalsIgnoreCase(name)) {
287                                    throw new ICalRawReader.StopReadingException();
288                            }
289    
290                            //find the component that this END property matches up with
291                            int popIndex = -1;
292                            for (int i = componentStack.size() - 1; i >= 0; i--) {
293                                    String n = componentNamesStack.get(i);
294                                    if (n.equalsIgnoreCase(name)) {
295                                            popIndex = i;
296                                            break;
297                                    }
298                            }
299                            if (popIndex == -1) {
300                                    //END property does not match up with any BEGIN properties, so ignore
301                                    addWarning("Ignoring END property that does not match up with any BEGIN properties: " + name, "END");
302                                    return;
303                            }
304    
305                            componentStack = componentStack.subList(0, popIndex);
306                            componentNamesStack = componentNamesStack.subList(0, popIndex);
307                    }
308    
309                    public void invalidLine(String line) {
310                            addWarning("Skipping malformed line: \"" + line + "\"");
311                    }
312    
313                    private ICalComponent getCurrentComponent() {
314                            if (componentStack.isEmpty()) {
315                                    return null;
316                            }
317                            return componentStack.get(componentStack.size() - 1);
318                    }
319            }
320    
321            private void addWarning(String message) {
322                    addWarning(message, null);
323            }
324    
325            private void addWarning(String message, String propertyName) {
326                    addWarning(message, propertyName, reader.getLineNum());
327            }
328    
329            private void addWarning(String message, String propertyName, int lineNum) {
330                    StringBuilder sb = new StringBuilder();
331                    sb.append("Line ").append(lineNum);
332                    if (propertyName != null) {
333                            sb.append(" (").append(propertyName).append(" property)");
334                    }
335                    sb.append(": ").append(message);
336    
337                    warnings.add(sb.toString());
338            }
339    
340            /**
341             * Closes the underlying {@link Reader} object.
342             */
343            public void close() throws IOException {
344                    reader.close();
345            }
