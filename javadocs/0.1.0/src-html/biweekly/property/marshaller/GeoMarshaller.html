035    /**
036     * Marshals {@link Geo} properties.
037     * @author Michael Angstadt
038     */
039    public class GeoMarshaller extends ICalPropertyMarshaller<Geo> {
040            public GeoMarshaller() {
041                    super(Geo.class, "GEO");
042            }
043    
044            @Override
045            protected String _writeText(Geo property) {
046                    ICalFloatFormatter formatter = new ICalFloatFormatter();
047                    StringBuilder sb = new StringBuilder();
048    
049                    if (property.getLatitude() != null) {
050                            sb.append(formatter.format(property.getLatitude()));
051                    }
052    
053                    sb.append(';');
054    
055                    if (property.getLongitude() != null) {
056                            sb.append(formatter.format(property.getLongitude()));
057                    }
058    
059                    return sb.toString();
060            }
061    
062            @Override
063            protected Geo _parseText(String value, ICalParameters parameters, List<String> warnings) {
064                    String split[] = value.split(";");
065    
066                    if (split.length < 2) {
067                            throw new CannotParseException("Could not parse value.");
068                    }
069    
070                    String latitudeStr = split[0];
071                    String longitudeStr = split[1];
072    
073                    Double latitude;
074                    try {
075                            latitude = Double.valueOf(latitudeStr);
076                    } catch (NumberFormatException e) {
077                            throw new CannotParseException("Could not parse latitude: " + latitudeStr);
078                    }
079    
080                    Double longitude;
081                    try {
082                            longitude = Double.valueOf(longitudeStr);
083                    } catch (NumberFormatException e) {
084                            throw new CannotParseException("Could not parse longtude: " + longitudeStr);
085                    }
086    
087                    return new Geo(latitude, longitude);
088            }
089    }
