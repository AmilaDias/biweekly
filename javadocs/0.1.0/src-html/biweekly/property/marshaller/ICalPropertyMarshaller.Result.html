041    /**
042     * Base class for iCalendar property marshallers.
043     * @author Michael Angstadt
044     */
045    public abstract class ICalPropertyMarshaller<T extends ICalProperty> {
046            private static final String NEWLINE = System.getProperty("line.separator");
047            protected final Class<T> clazz;
048            protected final String propertyName;
049    
050            /**
051             * Creates a new marshaller.
052             * @param clazz the property class
053             * @param propertyName the property name (e.g. "VERSION")
054             */
055            public ICalPropertyMarshaller(Class<T> clazz, String propertyName) {
056                    this.clazz = clazz;
057                    this.propertyName = propertyName.toUpperCase();
058            }
059    
060            /**
061             * Gets the property class.
062             * @return the property class
063             */
064            public Class<T> getPropertyClass() {
065                    return clazz;
066            }
067    
068            /**
069             * Gets the property name.
070             * @return the property name (e.g. "VERSION")
071             */
072            public String getPropertyName() {
073                    return propertyName;
074            }
075    
076            /**
077             * Sanitizes a property's parameters (called before the property is
078             * written). Note that a copy of the parameters is returned so that the
079             * property object does not get modified.
080             * @param property the property
081             * @return the sanitized parameters
082             */
083            public final ICalParameters prepareParameters(T property) {
084                    //make a copy because the property should not get modified when it is marshalled
085                    ICalParameters copy = new ICalParameters(property.getParameters());
086                    _prepareParameters(property, copy);
087                    return copy;
088            }
089    
090            /**
091             * Marshals a property's value to a string.
092             * @param property the property
093             * @return the marshalled value
094             */
095            public final String writeText(T property) {
096                    return _writeText(property);
097            }
098    
099            /**
100             * Unmarshals a property's value.
101             * @param value the value
102             * @param parameters the property's parameters
103             * @return the unmarshalled property object
104             * @throws CannotParseException if the marshaller could not parse the
105             * property's value
106             */
107            public final Result<T> parseText(String value, ICalParameters parameters) {
108                    List<String> warnings = new ArrayList<String>(0);
109                    T property = _parseText(value, parameters, warnings);
110                    property.setParameters(parameters);
111                    return new Result<T>(property, warnings);
112            }
113    
114            /**
115             * Sanitizes a property's parameters (called before the property is
116             * written). This should be overridden by child classes when required.
117             * @param property the property
118             * @param copy the list of parameters to make modifications to (it is a copy
119             * of the property's parameters)
120             */
121            protected void _prepareParameters(T property, ICalParameters copy) {
122                    //do nothing
123            }
124    
125            /**
126             * Marshals a property's value to a string.
127             * @param property the property
128             * @return the marshalled value
129             */
130            protected abstract String _writeText(T property);
131    
132            /**
133             * Unmarshals a property's value.
134             * @param value the value
135             * @param parameters the property's parameters
136             * @param warnings allows the programmer to alert the user to any
137             * note-worthy (but non-critical) issues that occurred during the
138             * unmarshalling process
139             * @return the unmarshalled property object
140             * @throws CannotParseException if the marshaller could not parse the
141             * property's value
142             */
143            protected abstract T _parseText(String value, ICalParameters parameters, List<String> warnings);
144    
145            /**
146             * Unescapes all special characters that are escaped with a backslash, as
147             * well as escaped newlines.
148             * @param text the text to unescape
149             * @return the unescaped text
150             */
151            protected static String unescape(String text) {
152                    StringBuilder sb = new StringBuilder(text.length());
153                    boolean escaped = false;
154                    for (int i = 0; i < text.length(); i++) {
155                            char ch = text.charAt(i);
156                            if (escaped) {
157                                    if (ch == 'n' || ch == 'N') {
158                                            //newlines appear as "\n" or "\N" (see RFC 2426 p.7)
159                                            sb.append(NEWLINE);
160                                    } else {
161                                            sb.append(ch);
162                                    }
163                                    escaped = false;
164                            } else if (ch == '\\') {
165                                    escaped = true;
166                            } else {
167                                    sb.append(ch);
168                            }
169                    }
170                    return sb.toString();
171            }
172    
173            /**
174             * Escapes all special characters within a iCalendar value.
175             * <p>
176             * These characters are:
177             * </p>
178             * <ul>
179             * <li>backslashes (<code>\</code>)</li>
180             * <li>commas (<code>,</code>)</li>
181             * <li>semi-colons (<code>;</code>)</li>
182             * <li>(newlines are escaped by {@link ICalWriter})</li>
183             * </ul>
184             * @param text the text to escape
185             * @return the escaped text
186             */
187            protected static String escape(String text) {
188                    String chars = "\\,;";
189                    for (int i = 0; i < chars.length(); i++) {
190                            String ch = chars.substring(i, i + 1);
191                            text = text.replace(ch, "\\" + ch);
192                    }
193                    return text;
194            }
195    
196            /**
197             * Splits a string by a character, taking escaped characters into account.
198             * Each split value is also trimmed.
199             * <p>
200             * Example:
201             * <p>
202             * <code>splitBy("HE\:LLO::WORLD", ':', false, true)</code>
203             * <p>
204             * returns
205             * <p>
206             * <code>["HE:LLO", "", "WORLD"]</code>
207             * @param str the string to split
208             * @param ch the character to split by
209             * @param removeEmpties true to remove empty elements, false not to
210             * @param unescape true to unescape each split string, false not to
211             * @return the split string
212             * @see <a
213             * href="http://stackoverflow.com/q/820172">http://stackoverflow.com/q/820172</a>
214             */
215            protected static String[] splitBy(String str, char ch, boolean removeEmpties, boolean unescape) {
216                    str = str.trim();
217                    String split[] = str.split("\\s*(?<!\\\\)" + Pattern.quote(ch + "") + "\\s*", -1);
218    
219                    List<String> list = new ArrayList<String>(split.length);
220                    for (String s : split) {
221                            if (s.length() == 0 && removeEmpties) {
222                                    continue;
223                            }
224    
225                            if (unescape) {
226                                    s = unescape(s);
227                            }
228    
229                            list.add(s);
230                    }
231    
232                    return list.toArray(new String[0]);
233            }
234    
235            /**
236             * Parses a comma-separated list of values.
237             * @param str the string to parse (e.g. "one,two,th\,ree")
238             * @return the parsed values
239             */
240            protected static String[] parseList(String str) {
241                    return splitBy(str, ',', true, true);
242            }
243    
244            /**
245             * Parses a component value.
246             * @param str the string to parse (e.g. "one;two,three;four")
247             * @return the parsed values
248             */
249            protected static String[][] parseComponent(String str) {
250                    String split[] = splitBy(str, ';', false, false);
251                    String ret[][] = new String[split.length][];
252                    int i = 0;
253                    for (String s : split) {
254                            String split2[] = parseList(s);
255                            ret[i++] = split2;
256                    }
257                    return ret;
258            }
259    
260            /**
261             * Parses a date or date-time string.
262             * @param value the date string
263             * @param timezoneId the TZID parameter value (or null if not set)
264             * @param warnings the warnings list
265             * @return the parsed date
266             * @throws IllegalArgumentException if the date string is invalid
267             */
268            protected static Date parseDate(String value, String timezoneId, List<String> warnings) {
269                    TimeZone timezone = null;
270                    if (timezoneId != null) {
271                            if (timezoneId.contains("/")) {
272                                    timezone = parseTimezoneId(timezoneId);
273                                    if (timezone == null) {
274                                            warnings.add("Ignoring unrecognized timezone ID: " + timezoneId);
275                                    }
276                            } else {
277                                    //TODO parse the date-time according to the referenced VTIMEZONE component
278                            }
279                    }
280    
281                    return ICalDateFormatter.parse(value, timezone);
282            }
283    
284            /**
285             * Writes a {@link Date} object as a string
286             * @param value the date
287             * @param hasTime whether the time component should be included
288             * @param timezoneId the TZID parameter value (or null if not set)
289             * @return the date string
290             */
291            protected static String writeDate(Date value, boolean hasTime, String timezoneId) {
292                    ISOFormat format;
293                    TimeZone timezone = null;
294    
295                    if (hasTime) {
296                            if (timezoneId == null) {
297                                    format = ISOFormat.UTC_TIME_BASIC;
298                            } else if (timezoneId.contains("/")) {
299                                    timezone = parseTimezoneId(timezoneId);
300                                    if (timezone == null) {
301                                            //unknown timezone
302                                            format = ISOFormat.UTC_TIME_BASIC;
303                                    } else {
304                                            format = ISOFormat.TIME_BASIC_WITHOUT_TZ;
305                                    }
306                            } else {
307                                    //TODO format the date-time according to the referenced VTIMEZONE component
308                                    format = ISOFormat.TIME_BASIC_WITHOUT_TZ;
309                            }
310                    } else {
311                            format = ISOFormat.DATE_BASIC;
312                    }
313    
314                    return ICalDateFormatter.format(value, format, timezone);
315            }
316    
317            /**
318             * Gets the {@link TimeZone} object that corresponds to the given ID.
319             * @param timezoneId the timezone ID (e.g. "America/New_York")
320             * @return the timezone object or null if not found
321             */
322            protected static TimeZone parseTimezoneId(String timezoneId) {
323                    TimeZone timezone = TimeZone.getTimeZone(timezoneId);
324                    return "GMT".equals(timezone.getID()) ? null : timezone;
325            }
326    
327            /**
328             * Represents the result of a marshal or unmarshal operation.
329             * @author Michael Angstadt
330             * @param <T> the marshalled/unmarshalled value (e.g. "String" if a property
331             * was marshalled)
332             */
333            public static class Result<T> {
334                    private final T value;
335                    private final List<String> warnings;
336    
337                    /**
338                     * Creates a new result.
339                     * @param value the value
340                     * @param warnings the warnings
341                     */
342                    public Result(T value, List<String> warnings) {
343                            this.value = value;
344                            this.warnings = warnings;
345                    }
346    
347                    /**
348                     * Gets the warnings.
349                     * @return the warnings
350                     */
351                    public List<String> getWarnings() {
352                            return warnings;
353                    }
354    
355                    /**
356                     * Gets the value.
357                     * @return the value
358                     */
359                    public T getValue() {
360                            return value;
361                    }
362            }
