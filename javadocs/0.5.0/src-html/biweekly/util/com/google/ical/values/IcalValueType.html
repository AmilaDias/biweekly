015package biweekly.util.com.google.ical.values;
016
017/**
018 * a value type for an Ical property.
019 *
020 * <pre>
021 * 4.2.20 Value Data Types
022 *
023 * Parameter Name: VALUE
024 *
025 * Purpose: To explicitly specify the data type format for a property value.
026 *
027 * Format Definition: The "VALUE" property parameter is defined by the
028 * following notation:
029 *
030 *      valuetypeparam = "VALUE" "=" valuetype
031 *
032 *      valuetype  = ("BINARY"
033 *                 / "BOOLEAN"
034 *                 / "CAL-ADDRESS"
035 *                 / "DATE"
036 *                 / "DATE-TIME"
037 *                 / "DURATION"
038 *                 / "FLOAT"
039 *                 / "INTEGER"
040 *                 / "PERIOD"
041 *                 / "RECUR"
042 *                 / "TEXT"
043 *                 / "TIME"
044 *                 / "URI"
045 *                 / "UTC-OFFSET"
046 *                 / x-name
047 *                 ; Some experimental iCalendar data type.
048 *                 / iana-token)
049 *                 ; Some other IANA registered iCalendar data type.
050 *
051 *
052 * Description: The parameter specifies the data type and format of the property
053 * value. The property values MUST be of a single value type. For example, a
054 * "RDATE" property cannot have a combination of DATE- TIME and TIME value
055 * types.
056 *
057 * If the property's value is the default value type, then this parameter need
058 * not be specified. However, if the property's default value type is overridden
059 * by some other allowable value type, then this parameter MUST be specified.
060 * </pre>
061 *
062 * @author mikesamuel+svn@gmail.com (Mike Samuel)
063 */
064public enum IcalValueType {
065
066  BINARY,
067  BOOLEAN,
068  CAL_ADDRESS,
069  DATE,
070  DATE_TIME,
071  DURATION,
072  FLOAT,
073  INTEGER,
074  PERIOD,
075  RECUR,
076  TEXT,
077  TIME,
078  URI,
079  UTC_OFFSET,
080
081  X_NAME,
082  OTHER,
083  ;
084
085  public static IcalValueType fromIcal(String icalValue) {
086    return valueOf(icalValue.toUpperCase().replace('-', '_'));
087  }
088
089  public String toIcal() { return name().replace('_', '-'); }
090}
