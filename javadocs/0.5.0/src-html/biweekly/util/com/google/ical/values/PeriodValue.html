015package biweekly.util.com.google.ical.values;
016
017
018/**
019 * a half-open range of {@link DateValue}s.  The start is inclusive, and the
020 * end is exclusive.  The end must be on or after the start.  When the start and
021 * end are the same, the period is zero width, i.e. contains zero seconds.
022 *
023 * @author mikesamuel+svn@gmail.com (Mike Samuel)
024 */
025public interface PeriodValue {
026
027  /**
028   * the start of the period.
029   * @return non null.
030   */
031  DateValue start();
032
033  /**
034   * the end of the period.
035   * The end must be &gt;= {@link #start()}, and
036   * <tt>(start() instanceof {@link TimeValue}) ==
037   *     (end() instanceof TimeValue)</tt>.
038   * @return non null.
039   */
040  DateValue end();
041
042  /**
043   * true iff this period overlaps the given period.
044   * @param pv not null.
045   */
046  boolean intersects(PeriodValue pv);
047
048  /**
049   * true iff this period completely contains the given period.
050   * @param pv not null.
051   */
052  boolean contains(PeriodValue pv);
053
054}  // PeriodValue
