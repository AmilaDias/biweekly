015package biweekly.util.com.google.ical.values;
016
017import biweekly.util.com.google.ical.util.DTBuilder;
018import biweekly.util.com.google.ical.util.TimeUtils;
019
020import java.text.ParseException;
021import java.util.TimeZone;
022import java.util.regex.Matcher;
023import java.util.regex.Pattern;
024
025/**
026 * static functions for parsing ical values.
027 *
028 * @author mikesamuel+svn@gmail.com (Mike Samuel)
029 */
030public final class IcalParseUtil {
031
032  private static final Pattern DATE_VALUE = Pattern.compile(
033      "(\\d{4,})(\\d\\d)(\\d\\d)" +
034      "(?:T([0-1]\\d|2[0-3])([0-5]\\d)([0-5]\\d)(Z)?)?");
035
036  /** parses a date of the form yyyymmdd or yyyymmdd'T'hhMMss */
037  public static DateValue parseDateValue(String s) throws ParseException {
038    return parseDateValue(s, null);
039  }
040
041  /**
042   * parses a date of the form yyyymmdd or yyyymmdd'T'hhMMss converting from
043   * the given timezone to UTC.
044   */
045  public static DateValue parseDateValue(String s, TimeZone tzid)
046      throws ParseException {
047    Matcher m = DATE_VALUE.matcher(s);
048    if (!m.matches()) { throw new ParseException(s, 0); }
049    int year = Integer.parseInt(m.group(1)),
050      month = Integer.parseInt(m.group(2)),
051      day = Integer.parseInt(m.group(3));
052    if (null != m.group(4)) {
053      int hour = Integer.parseInt(m.group(4)),
054        minute = Integer.parseInt(m.group(5)),
055        second = Integer.parseInt(m.group(6));
056      boolean utc = null != m.group(7);
057
058      DateValue dv = new DTBuilder(
059          year, month, day, hour, minute, second).toDateTime();
060      if (!utc && null != tzid) {
061        dv = TimeUtils.toUtc(dv, tzid);
062      }
063      return dv;
064    } else {
065      return new DTBuilder(year, month, day).toDate();
066    }
067  }
068
069  /**
070   * parse a period value of the form &lt;start&gt;/&lt;end&gt;.
071   * This does not yet recognize the &lt;start&gt;/&lt;duration&gt; form.
072   */
073  public static PeriodValue  parsePeriodValue(String s) throws ParseException {
074    return parsePeriodValue(s, null);
075  }
076
077  /**
078   * parse a period value of the form &lt;start&gt;/&lt;end&gt;, converting
079   * from the given timezone to UTC.
080   * This does not yet recognize the &lt;start&gt;/&lt;duration&gt; form.
081   */
082  public static PeriodValue  parsePeriodValue(String s, TimeZone tzid)
083      throws ParseException {
084    int sep = s.indexOf('/');
085    if (sep < 0) { throw new ParseException(s, s.length()); }
086    DateValue start = parseDateValue(s.substring(0, sep), tzid),
087                end = parseDateValue(s.substring(sep + 1), tzid);
088    if ((start instanceof TimeValue) != (end instanceof TimeValue)) {
089      throw new ParseException(s, 0);
090    }
091    try {
092      return PeriodValueImpl.create(start, end);
093    } catch (IllegalArgumentException ex) {
094      throw (ParseException) new ParseException(s, sep + 1).initCause(ex);
095    }
096  }
097
098  /**
099   * unfolds ical content lines as per RFC 2445 section 4.1.
100   *
101   * <h3>4.1 Content Lines</h3>
102   *
103   * <p>The iCalendar object is organized into individual lines of text, called
104   * content lines. Content lines are delimited by a line break, which is a CRLF
105   * sequence (US-ASCII decimal 13, followed by US-ASCII decimal 10).
106   *
107   * <p>Lines of text SHOULD NOT be longer than 75 octets, excluding the line
108   * break. Long content lines SHOULD be split into a multiple line
109   * representations using a line "folding" technique. That is, a long line can
110   * be split between any two characters by inserting a CRLF immediately
111   * followed by a single linear white space character (i.e., SPACE, US-ASCII
112   * decimal 32 or HTAB, US-ASCII decimal 9). Any sequence of CRLF followed
113   * immediately by a single linear white space character is ignored (i.e.,
114   * removed) when processing the content type.
115   */
116  public static String unfoldIcal(String foldedContentLines) {
117    return IGNORABLE_ICAL_WHITESPACE.matcher(foldedContentLines).replaceAll("");
118  }
119  private static final Pattern IGNORABLE_ICAL_WHITESPACE =
120    Pattern.compile("(?:\\r\\n?|\\n)[ \t]");
121
122  private IcalParseUtil() {
123    // uninstantiable.
124  }
125}
