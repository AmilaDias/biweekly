042/**
043 * for calculating the occurrences of an individual RFC 2445 RRULE or groups of
044 * RRULES, RDATES, EXRULES, and EXDATES.
045 *
046 * <h4>Glossary</h4>
047 * Period - year|month|day|...<br>
048 * Day of the week - an int in [0-6].  See RRULE_WDAY_* in rrule.js<br>
049 * Day of the year - zero indexed in [0,365]<br>
050 * Day of the month - 1 indexed in [1,31]<br>
051 * Month - 1 indexed integer in [1,12]
052 *
053 * <h4>Abstractions</h4>
054 * Generator - a function corresponding to an RRULE part that takes a date and
055 *   returns a later (year or month or day depending on its period) within the
056 *   next larger period.
057 *   A generator ignores all periods in its input smaller than its period.
058 * <p>
059 * Filter - a function that returns true iff the given date matches the subrule.
060 * <p>
061 * Condition - returns true if the given date is past the end of the recurrence.
062 *
063 * <p>All the functions that represent rule parts are stateful.
064 *
065 * @author mikesamuel+svn@gmail.com (Mike Samuel)
066 */
067public class RecurrenceIteratorFactory {
068
069  private static final Logger LOGGER = Logger.getLogger(
070      RecurrenceIteratorFactory.class.getName());
071
072  /**
073   * given a block of RRULE, EXRULE, RDATE, and EXDATE content lines, parse
074   * them into a single recurrence iterator.
075   * @param rdata ical text.
076   * @param dtStart the date of the first occurrence in timezone tzid, which is
077   *   used to fill in optional fields in the RRULE, such as the day of the
078   *   month for a monthly repetition when no ther day specified.
079   *   Note: this may not be the first date in the series since an EXRULE or
080   *     EXDATE might force it to be skipped, but there will be no earlier date
081   *     generated by this ruleset.
082   * @param strict true if any failure to parse should result in a
083   *   ParseException.  false causes bad content lines to be logged and ignored.
084   */
085  public static RecurrenceIterator createRecurrenceIterator(
086      String rdata, DateValue dtStart, TimeZone tzid, boolean strict)
087      throws ParseException {
088    return createRecurrenceIterable(rdata, dtStart, tzid, strict).iterator();
089  }
090
091  public static RecurrenceIterable createRecurrenceIterable(
092      String rdata, final DateValue dtStart, final TimeZone tzid,
093      final boolean strict)
094      throws ParseException {
095    final IcalObject[] contentLines = parseContentLines(rdata, tzid, strict);
096
097    return new RecurrenceIterable() {
098        public RecurrenceIterator iterator() {
099          List<RecurrenceIterator> inclusions =
100               new ArrayList<RecurrenceIterator>();
101          List<RecurrenceIterator> exclusions =
102               new ArrayList<RecurrenceIterator>();
103          // always include DTStart
104          inclusions.add(new RDateIteratorImpl(
105                             new DateValue[] {TimeUtils.toUtc(dtStart, tzid)}));
106          for (IcalObject contentLine : contentLines) {
107            try {
108              String name = contentLine.getName();
109              if ("rrule".equalsIgnoreCase(name)) {
110                inclusions.add(createRecurrenceIterator(
111                                   (RRule) contentLine, dtStart, tzid));
112              } else if ("rdate".equalsIgnoreCase(name)) {
113                inclusions.add(
114                    createRecurrenceIterator((RDateList) contentLine));
115              } else if ("exrule".equalsIgnoreCase(name)) {
116                exclusions.add(createRecurrenceIterator(
117                                   (RRule) contentLine, dtStart, tzid));
118              } else if ("exdate".equalsIgnoreCase(name)) {
119                exclusions.add(
120                    createRecurrenceIterator((RDateList) contentLine));
121              }
122            } catch (IllegalArgumentException ex) {
123              // bad frequency on rrule or exrule
124              if (strict) { throw ex; }
125              LOGGER.log(
126                  Level.SEVERE,
127                  "Dropping bad recurrence rule line: " + contentLine.toIcal(),
128                  ex);
129            }
130          }
131          return new CompoundIteratorImpl(inclusions, exclusions);
132        }
133      };
134  }

146  /**
147   * create a recurrence iterator from an rdate or exdate list.
148   */
149  public static RecurrenceIterator createRecurrenceIterator(RDateList rdates) {
150    DateValue[] dates = rdates.getDatesUtc();
151    Arrays.sort(dates);
152    int k = 0;
153    for (int i = 1; i < dates.length; ++i) {
154      if (!dates[i].equals(dates[k])) { dates[++k] = dates[i]; }
155    }
156    if (++k < dates.length) {
157      DateValue[] uniqueDates = new DateValue[k ];
158      System.arraycopy(dates, 0, uniqueDates, 0, k);
159      dates = uniqueDates;
160    }
161    return new RDateIteratorImpl(dates);
162  }

164  /**
165   * create a recurrence iterator from an rrule.
166   * @param rrule the recurrence rule to iterate.
167   * @param dtStart the start of the series, in tzid.
168   * @param tzid the timezone to iterate in.
169   */
170  public static RecurrenceIterator createRecurrenceIterator(
171      RRule rrule, DateValue dtStart, TimeZone tzid) {
172    assert null != tzid;
173    assert null != dtStart;
174
175    Frequency freq = rrule.getFreq();
176    Weekday wkst = rrule.getWkSt();
177    DateValue untilUtc = rrule.getUntil();
178    int count = rrule.getCount();
179    int interval = rrule.getInterval();
180    WeekdayNum[] byDay = rrule.getByDay().toArray(new WeekdayNum[0]);
181    int[] byMonth = rrule.getByMonth();
182    int[] byMonthDay = rrule.getByMonthDay();
183    int[] byWeekNo = rrule.getByWeekNo();
184    int[] byYearDay = rrule.getByYearDay();
185    int[] bySetPos = rrule.getBySetPos();
186    int[] byHour = rrule.getByHour();
187    int[] byMinute = rrule.getByMinute();
188    int[] bySecond = rrule.getBySecond();
189
190    if (interval <= 0) {  interval = 1; }
191
192    if (null == wkst) {
193      wkst = Weekday.MO;
194    }
195
196    // Optimize out BYSETPOS where possible.
197    if (bySetPos.length != 0) {
198      switch (freq) {
199        case HOURLY:
200          // ;BYHOUR=3,6,9;BYSETPOS=-1,1
201          //     is equivalent to
202          // ;BYHOUR=3,9
203          if (byHour.length != 0 && byMinute.length <= 1
204              && bySecond.length <= 1) {
205            byHour = filterBySetPos(byHour, bySetPos);
206          }
207          // Handling bySetPos for rules that are more frequent than daily
208          // tends to lead to large amounts of processor being used before other
209          // work limiting features can kick in since there many seconds between
210          // dtStart and where the year limit kicks in.
211          // There are no known use cases for the use of bySetPos with hourly
212          // minutely and secondly rules so we just ignore it.
213          bySetPos = NO_INTS;
214          break;
215        case MINUTELY:
216          // ;BYHOUR=3,6,9;BYSETPOS=-1,1
217          //     is equivalent to
218          // ;BYHOUR=3,9
219          if (byMinute.length != 0 && bySecond.length <= 1) {
220            byMinute = filterBySetPos(byMinute, bySetPos);
221          }
222          // See bySetPos handling comment above.
223          bySetPos = NO_INTS;
224          break;
225        case SECONDLY:
226          // ;BYHOUR=3,6,9;BYSETPOS=-1,1
227          //     is equivalent to
228          // ;BYHOUR=3,9
229          if (bySecond.length != 0) {
230            bySecond = filterBySetPos(bySecond, bySetPos);
231          }
232          // See bySetPos handling comment above.
233          bySetPos = NO_INTS;
234          break;
235        default:
236      }
237    }
238
239    DateValue start = dtStart;
240    if (bySetPos.length != 0) {
241      // Roll back till the beginning of the period to make sure that any
242      // positive indices are indexed properly.
243      // The actual iterator implementation is responsible for anything
244      // < dtStart.
245      switch (freq) {
246        case YEARLY:
247          start = dtStart instanceof TimeValue
248              ? new DateTimeValueImpl(start.year(), 1, 1, 0, 0, 0)
249              : new DateValueImpl(start.year(), 1, 1);
250          break;
251        case MONTHLY:
252          start = dtStart instanceof TimeValue
253              ? new DateTimeValueImpl(start.year(), start.month(), 1, 0, 0, 0)
254              : new DateValueImpl(start.year(), start.month(), 1);
255          break;
256        case WEEKLY:
257          int d = (7 + wkst.ordinal() - Weekday.valueOf(dtStart).ordinal()) % 7;
258          start = TimeUtils.add(dtStart, new DateValueImpl(0, 0, -d));
259          break;
260        default: break;
261      }
262    }
263
264    // recurrences are implemented as a sequence of periodic generators.
265    // First a year is generated, and then months, and within months, days
266    ThrottledGenerator yearGenerator = Generators.serialYearGenerator(
267        freq == Frequency.YEARLY ? interval : 1, dtStart);
268    Generator monthGenerator = null;
269    Generator dayGenerator = null;
270    Generator secondGenerator = null;
271    Generator minuteGenerator = null;
272    Generator hourGenerator = null;
273
274    // When multiple generators are specified for a period, they act as a union
275    // operator.  We could have multiple generators (for day say) and then
276    // run each and merge the results, but some generators are more efficient
277    // than others, so to avoid generating 53 sundays and throwing away all but
278    // 1 for RRULE:FREQ=YEARLY;BYDAY=TU;BYWEEKNO=1, we reimplement some of the
279    // more prolific generators as filters.
280    // TODO(msamuel): don't need a list here
281    List<Predicate<? super DateValue>> filters =
282      new ArrayList<Predicate<? super DateValue>>();
283
284    switch (freq) {
285      case SECONDLY:
286        if (bySecond.length == 0 || interval != 1) {
287          secondGenerator = Generators.serialSecondGenerator(interval, dtStart);
288          if (bySecond.length != 0) {
289            filters.add(Filters.bySecondFilter(bySecond));
290          }
291        }
292        break;
293      case MINUTELY:
294        if (byMinute.length == 0 || interval != 1) {
295          minuteGenerator = Generators.serialMinuteGenerator(interval, dtStart);
296          if (byMinute.length != 0) {
297            filters.add(Filters.byMinuteFilter(byMinute));
298          }
299        }
300        break;
301      case HOURLY:
302        if (byHour.length == 0 || interval != 1) {
303          hourGenerator = Generators.serialHourGenerator(interval, dtStart);
304          if (byHour.length != 0) {
305            filters.add(Filters.byHourFilter(bySecond));
306          }
307        }
308        break;
309      case DAILY:
310        break;
311      case WEEKLY:
312        // week is not considered a period because a week may span multiple
313        // months &| years.  There are no week generators, but so a filter is
314        // used to make sure that FREQ=WEEKLY;INTERVAL=2 only generates dates
315        // within the proper week.
316        if (0 != byDay.length) {
317          dayGenerator = Generators.byDayGenerator(byDay, false, start);
318          byDay = NO_DAYS;
319          if (interval > 1) {
320            filters.add(Filters.weekIntervalFilter(interval, wkst, dtStart));
321          }
322        } else {
323          dayGenerator = Generators.serialDayGenerator(interval * 7, dtStart);
324        }
325        break;
326      case YEARLY:
327        if (0 != byYearDay.length) {
328          // The BYYEARDAY rule part specifies a COMMA separated list of days of
329          // the year. Valid values are 1 to 366 or -366 to -1. For example, -1
330          // represents the last day of the year (December 31st) and -306
331          // represents the 306th to the last day of the year (March 1st).
332          dayGenerator = Generators.byYearDayGenerator(byYearDay, start);
333          break;
334        }
335        // $FALL-THROUGH$
336      case MONTHLY:
337        if (0 != byMonthDay.length) {
338          // The BYMONTHDAY rule part specifies a COMMA separated list of days
339          // of the month. Valid values are 1 to 31 or -31 to -1. For example,
340          // -10 represents the tenth to the last day of the month.
341          dayGenerator = Generators.byMonthDayGenerator(byMonthDay, start);
342          byMonthDay = NO_INTS;
343        } else if (0 != byWeekNo.length && Frequency.YEARLY == freq) {
344          // The BYWEEKNO rule part specifies a COMMA separated list of ordinals
345          // specifying weeks of the year.  This rule part is only valid for
346          // YEARLY rules.
347          dayGenerator = Generators.byWeekNoGenerator(byWeekNo, wkst, start);
348          byWeekNo = NO_INTS;
349        } else if (0 != byDay.length) {
350          // Each BYDAY value can also be preceded by a positive (n) or negative
351          // (-n) integer. If present, this indicates the nth occurrence of the
352          // specific day within the MONTHLY or YEARLY RRULE. For example,
353          // within a MONTHLY rule, +1MO (or simply 1MO) represents the first
354          // Monday within the month, whereas -1MO represents the last Monday of
355          // the month. If an integer modifier is not present, it means all days
356          // of this type within the specified frequency. For example, within a
357          // MONTHLY rule, MO represents all Mondays within the month.
358          dayGenerator = Generators.byDayGenerator(
359              byDay, Frequency.YEARLY == freq && 0 == byMonth.length, start);
360          byDay = NO_DAYS;
361        } else {
362          if (Frequency.YEARLY == freq) {
363            monthGenerator = Generators.byMonthGenerator(
364                new int[] { dtStart.month() }, start);
365          }
366          dayGenerator = Generators.byMonthDayGenerator(
367              new int[] { dtStart.day() }, start);
368        }
369        break;
370    }
371
372    if (secondGenerator == null) {
373      secondGenerator = Generators.bySecondGenerator(bySecond, start);
374    }
375    if (minuteGenerator == null) {
376      if (byMinute.length == 0 && freq.compareTo(Frequency.MINUTELY) < 0) {
377        minuteGenerator = Generators.serialMinuteGenerator(1, dtStart);
378      } else {
379        minuteGenerator = Generators.byMinuteGenerator(byMinute, start);
380      }
381    }
382    if (hourGenerator == null) {
383      if (byHour.length == 0 && freq.compareTo(Frequency.HOURLY) < 0) {
384        hourGenerator = Generators.serialHourGenerator(1, dtStart);
385      } else {
386        hourGenerator = Generators.byHourGenerator(byHour, start);
387      }
388    }
389
390    if (dayGenerator == null) {
391      boolean dailyOrMoreOften = freq.compareTo(Frequency.DAILY) <= 0;
392      if (byMonthDay.length != 0) {
393        dayGenerator = Generators.byMonthDayGenerator(byMonthDay, start);
394        byMonthDay = NO_INTS;
395      } else if (byDay.length != 0) {
396        dayGenerator = Generators.byDayGenerator(
397            byDay, Frequency.YEARLY == freq, start);
398        byDay = NO_DAYS;
399      } else if (dailyOrMoreOften) {
400        dayGenerator = Generators.serialDayGenerator(
401            Frequency.DAILY == freq ? interval : 1, dtStart);
402      } else {
403        dayGenerator = Generators.byMonthDayGenerator(
404            new int[] { dtStart.day() }, start);
405      }
406    }
407
408    if (0 != byDay.length) {
409      filters.add(Filters.byDayFilter(byDay, Frequency.YEARLY == freq, wkst));
410      byDay = NO_DAYS;
411    }
412
413    if (0 != byMonthDay.length) {
414      filters.add(Filters.byMonthDayFilter(byMonthDay));
415    }
416
417    // generator inference common to all periods
418    if (0 != byMonth.length) {
419      monthGenerator = Generators.byMonthGenerator(byMonth, start);
420    } else if (null == monthGenerator) {
421      monthGenerator = Generators.serialMonthGenerator(
422          freq == Frequency.MONTHLY ? interval : 1, dtStart);
423    }
424
425    // the condition tells the iterator when to halt.
426    // The condition is exclusive, so the date that triggers it will not be
427    // included.
428    Predicate<DateValue> condition;
429    boolean canShortcutAdvance = true;
430    if (0 != count) {
431      condition = Conditions.countCondition(count);
432      // We can't shortcut because the countCondition must see every generated
433      // instance.
434      // TODO(msamuel): if count is large, we might try predicting the end date
435      // so that we can convert the COUNT condition to an UNTIL condition.
436      canShortcutAdvance = false;
437    } else if (null != untilUtc) {
438      if ((untilUtc instanceof TimeValue) != (dtStart instanceof TimeValue)) {
439        // TODO(msamuel): warn
440        if (dtStart instanceof TimeValue) {
441          untilUtc = TimeUtils.dayStart(untilUtc);
442        } else {
443          untilUtc = TimeUtils.toDateValue(untilUtc);
444        }
445      }
446      condition = Conditions.untilCondition(untilUtc);
447    } else {
448      condition = Predicates.<DateValue>alwaysTrue();
449    }
450
451    // combine filters into a single function
452    Predicate<? super DateValue> filter;
453    switch (filters.size()) {
454      case 0:
455        filter = Predicates.<DateValue>alwaysTrue();
456        break;
457      case 1:
458        filter = filters.get(0);
459        break;
460      default:
461        filter = Predicates.and(filters);
462        break;
463    }
464
465    Generator instanceGenerator;
466    if (0 != bySetPos.length) {
467      instanceGenerator = InstanceGenerators.bySetPosInstanceGenerator(
468          bySetPos, freq, wkst, filter,
469          yearGenerator, monthGenerator, dayGenerator, hourGenerator,
470          minuteGenerator, secondGenerator);
471    } else {
472      instanceGenerator = InstanceGenerators.serialInstanceGenerator(
473          filter, yearGenerator, monthGenerator, dayGenerator,
474          hourGenerator, minuteGenerator, secondGenerator);
475    }
476
477    return new RRuleIteratorImpl(
478        dtStart, tzid, condition, instanceGenerator,
479        yearGenerator, monthGenerator, dayGenerator,
480        hourGenerator, minuteGenerator, secondGenerator,
481        canShortcutAdvance);
482  }

484  /**
485   * a recurrence iterator that returns the union of the given recurrence
486   * iterators.
487   */
488  public static RecurrenceIterator join(
489      RecurrenceIterator a, RecurrenceIterator... b) {
490    List<RecurrenceIterator> incl = new ArrayList<RecurrenceIterator>();
491    incl.add(a);
492    incl.addAll(Arrays.asList(b));
493    return new CompoundIteratorImpl(
494        incl, Collections.<RecurrenceIterator>emptyList());
495  }

497  /**
498   * an iterator over all the dates included except those excluded, i.e.
499   * <code>inclusions - exclusions</code>.
500   * Exclusions trump inclusions, and {@link DateValue dates} and
501   * {@link DateTimeValue date-times} never match one another.
502   * @param included non null.
503   * @param excluded non null.
504   * @return non null.
505   */
506  public static RecurrenceIterator except(
507      RecurrenceIterator included, RecurrenceIterator excluded) {
508    return new CompoundIteratorImpl(
509        Collections.<RecurrenceIterator>singleton(included),
510        Collections.<RecurrenceIterator>singleton(excluded));
511  }

513  private static final Pattern FOLD = Pattern.compile("(?:\\r\\n?|\\n)[ \t]");
514  private static final Pattern NEWLINE = Pattern.compile("[\\r\\n]+");
515  private static final Pattern RULE = Pattern.compile(
516      "^(?:R|EX)RULE[:;]", Pattern.CASE_INSENSITIVE);
517  private static final Pattern DATE = Pattern.compile(
518      "^(?:R|EX)DATE[:;]", Pattern.CASE_INSENSITIVE);
519  private static IcalObject[] parseContentLines(
520      String rdata, TimeZone tzid, boolean strict)
521      throws ParseException {
522    String unfolded = FOLD.matcher(rdata).replaceAll("").trim();
523    if ("".equals(unfolded)) { return new IcalObject[0]; }
524    String[] lines = NEWLINE.split(unfolded);
525    IcalObject[] out = new IcalObject[lines.length];
526    int nbad = 0;
527    for (int i = 0; i < lines.length; ++i) {
528      String line = lines[i].trim();
529      try {
530        if (RULE.matcher(line).find()) {
531          out[i] = new RRule(line);
532        } else if (DATE.matcher(line).find()) {
533          out[i] = new RDateList(line, tzid);
534        } else {
535          throw new ParseException(lines[i], i);
536        }
537      } catch (ParseException ex) {
538        if (strict) {
539          throw ex;
540        }
541        LOGGER.log(Level.SEVERE,
542                   "Dropping bad recurrence rule line: " + line, ex);
543        ++nbad;
544      } catch (IllegalArgumentException ex) {
545        if (strict) {
546          throw ex;
547        }
548        LOGGER.log(Level.SEVERE,
549                   "Dropping bad recurrence rule line: " + line, ex);
550        ++nbad;
551      }
552    }
553    if (0 != nbad) {
554      IcalObject[] trimmed = new IcalObject[out.length - nbad];
555      for (int i = 0, k = 0; i < trimmed.length; ++k) {
556        if (null != out[k]) { trimmed[i++] = out[k]; }
557      }
558      out = trimmed;
559    }
560    return out;
561  }

563  /**
564   * Given an array like BYMONTH=2,3,4,5 and a set pos like BYSETPOS=1,-1
565   * reduce both clauses to a single one, BYMONTH=2,5 in the preceding.
566   */
567  private static int[] filterBySetPos(int[] members, int[] bySetPos) {
568    members = Util.uniquify(members);
569    IntSet iset = new IntSet();
570    for (int pos : bySetPos) {
571      if (pos == 0) { continue; }
572      if (pos < 0) {
573        pos += members.length;
574      } else {
575        --pos;  // Zero-index.
576      }
577      if (pos >= 0 && pos < members.length) {
578        iset.add(members[pos]);
579      }
580    }
581    return iset.toIntArray();
582  }
