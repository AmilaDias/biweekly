015package biweekly.util.com.google.ical.values;
016
017/**
018 * represents a day of the week in a month or year such as the
019 * third monday of the month.  A negative num indicates it counts from the
020 * end of the month or year.
021 *
022 * <blockquote>
023 * Each BYDAY value can also be preceded by a positive (+n) or negative
024 * (-n) integer. If present, this indicates the nth occurrence of the
025 * specific day within the MONTHLY or YEARLY RRULE. For example, within
026 * a MONTHLY rule, +1MO (or simply 1MO) represents the first Monday
027 * within the month, whereas -1MO represents the last Monday of the
028 * month. If an integer modifier is not present, it means all days of
029 * this type within the specified frequency. For example, within a
030 * MONTHLY rule, MO represents all Mondays within the month.
031 * </blockquote>
032 *
033 * @author mikesamuel+svn@gmail.com (Mike Samuel)
034 */
035public class WeekdayNum {
036  public final int num;
037  public final Weekday wday;
038
039  /**
040   * @param num in -53,53
041   * @param wday non null.
042   */
043  public WeekdayNum(int num, Weekday wday) {
044    if (!(-53 <= num && 53 >= num && null != wday)) {
045      throw new IllegalArgumentException();
046    }
047    this.num = num;
048    this.wday = wday;
049  }
050  public String toIcal() {
051    return (0 != this.num)
052        ? String.valueOf(this.num) + this.wday
053        : this.wday.toString();
054  }
055  @Override
056  public String toString() {
057    return toIcal();
058  }
059  @Override
060  public boolean equals(Object o) {
061    if (!(o instanceof WeekdayNum)) { return false; }
062    WeekdayNum wdn = (WeekdayNum) o;
063    return this.num == wdn.num && this.wday == wdn.wday;
064  }
065
066  @Override
067  public int hashCode() {
068    return num ^ (53 * wday.hashCode());
069  }
070}
