045/**
046 * Writes data to an iCalendar data stream.
047 * @author Michael Angstadt
048 * @see <a href="http://www.imc.org/pdi/pdiproddev.html">1.0 specs</a>
049 * @see <a href="https://tools.ietf.org/html/rfc2445">RFC 2445</a>
050 * @see <a href="http://tools.ietf.org/html/rfc5545">RFC 5545</a>
051 */
052public class ICalRawWriter implements Closeable, Flushable {
053        /**
054         * If any of these characters are found within a parameter value, then the
055         * entire parameter value must be wrapped in double quotes (applies to
056         * version 2.0 only).
057         */
058        private final CharacterBitSet specialParameterCharacters = new CharacterBitSet(",:;");
059
060        /**
061         * Regular expression used to detect newline character sequences.
062         */
063        private final Pattern newlineRegex = Pattern.compile("\\r\\n|\\r|\\n");
064        private final CharacterBitSet newlineBitSet = new CharacterBitSet("\r\n");
065
066        /**
067         * List of characters which would break the syntax of the iCalendar object
068         * if used inside a property name. The list of characters permitted by the
069         * specification is much more strict, but the goal here is to be as lenient
070         * as possible.
071         */
072        private final CharacterBitSet invalidPropertyNameCharacters = new CharacterBitSet(";:\n\r");
073
074        /**
075         * List of characters which would break the syntax of the iCalendar object
076         * if used inside a parameter value with caret encoding disabled. These
077         * characters cannot be escaped or encoded, so they are impossible to
078         * include inside of a parameter value. The list of characters permitted by
079         * the specification is much more strict, but the goal here is to be as
080         * lenient as possible.
081         */
082        private final Map<ICalVersion, CharacterBitSet> invalidParamValueChars;
083        {
084                Map<ICalVersion, CharacterBitSet> map = new EnumMap<ICalVersion, CharacterBitSet>(ICalVersion.class);
085
086                map.put(ICalVersion.V1_0, new CharacterBitSet(",:\n\r")); //note: semicolons can be escaped
087                map.put(ICalVersion.V2_0_DEPRECATED, new CharacterBitSet("\"\r\n"));
088                map.put(ICalVersion.V2_0, map.get(ICalVersion.V2_0_DEPRECATED));
089
090                invalidParamValueChars = Collections.unmodifiableMap(map);
091        }
092
093        /**
094         * List of characters which would break the syntax of the iCalendar object
095         * if used inside a parameter value with caret encoding enabled. These
096         * characters cannot be escaped or encoded, so they are impossible to
097         * include inside of a parameter value. The list of characters permitted by
098         * the specification is much more strict, but the goal here is to be as
099         * lenient as possible.
100         */
101        private final Map<ICalVersion, CharacterBitSet> invalidParamValueCharsWithCaretEncoding;
102        {
103                Map<ICalVersion, CharacterBitSet> map = new EnumMap<ICalVersion, CharacterBitSet>(ICalVersion.class);
104
105                map.put(ICalVersion.V1_0, invalidParamValueChars.get(ICalVersion.V1_0)); //1.0 does not support caret encoding
106                map.put(ICalVersion.V2_0_DEPRECATED, new CharacterBitSet(""));
107                map.put(ICalVersion.V2_0, map.get(ICalVersion.V2_0_DEPRECATED));
108
109                invalidParamValueCharsWithCaretEncoding = Collections.unmodifiableMap(map);
110        }
111
112        private final FoldedLineWriter writer;
113        private boolean caretEncodingEnabled = false;
114        private ICalVersion version;

116        /**
117         * @param writer the writer to wrap
118         * @param version the version to adhere to
119         */
120        public ICalRawWriter(Writer writer, ICalVersion version) {
121                this.writer = new FoldedLineWriter(writer);
122                this.version = version;
123        }

126         * Gets the writer that this object wraps.

133        /**
134         * <p>
135         * Gets whether the writer will apply circumflex accent encoding on
136         * parameter values (disabled by default, only applies to version 2.0). This
137         * escaping mechanism allows for newlines and double quotes to be included
138         * in parameter values.
139         * </p>
140         * 
141         * <p>
142         * Note that this encoding mechanism is defined separately from the
143         * iCalendar specification and may not be supported by the iCalendar
144         * consumer.
145         * </p>
146         * 
147         * <table class="simpleTable">
148         * <caption>Characters encoded by circumflex accent encoding</caption>
149         * <tr>
150         * <th>Raw Character</th>
151         * <th>Encoded Character</th>
152         * </tr>
153         * <tr>
154         * <td>{@code "}</td>
155         * <td>{@code ^'}</td>
156         * </tr>
157         * <tr>
158         * <td><i>newline</i></td>
159         * <td>{@code ^n}</td>
160         * </tr>
161         * <tr>
162         * <td>{@code ^}</td>
163         * <td>{@code ^^}</td>
164         * </tr>
165         * </table>
166         * 
167         * <p>
168         * Example:
169         * </p>
170         * 
171         * <pre>
172         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPittsburgh, PA 15212":40.446816;80.00566
173         * </pre>
174         * 
175         * @return true if circumflex accent encoding is enabled, false if not
176         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
177         */
178        public boolean isCaretEncodingEnabled() {
179                return caretEncodingEnabled;
180        }

188         * </p>
189         * 
190         * <p>
191         * Note that this encoding mechanism is defined separately from the
192         * iCalendar specification and may not be supported by the iCalendar
193         * consumer.
194         * </p>
195         * 
196         * <table class="simpleTable">
197         * <caption>Characters encoded by circumflex accent encoding</caption>
198         * <tr>
199         * <th>Raw Character</th>
200         * <th>Encoded Character</th>
201         * </tr>
202         * <tr>
203         * <td>{@code "}</td>

274        /**
275         * Writes a property to the iCalendar data stream.
276         * @param propertyName the property name (e.g. "SUMMARY")
277         * @param value the property value
278         * @throws IllegalArgumentException if the property data contains one or
279         * more characters which break the iCalendar syntax and which cannot be
280         * escaped or encoded
281         * @throws IOException if there's a problem writing to the data stream
282         */
283        public void writeProperty(String propertyName, String value) throws IOException {
284                writeProperty(propertyName, new ICalParameters(), value);
285        }
286
287        /**
288         * Writes a property to the iCalendar data stream.
289         * @param propertyName the property name (e.g. "SUMMARY")
290         * @param parameters the property parameters
291         * @param value the property value
292         * @throws IllegalArgumentException if the property data contains one or
293         * more characters which break the iCalendar syntax and which cannot be
294         * escaped or encoded
295         * @throws IOException if there's a problem writing to the data stream
296         */
297        public void writeProperty(String propertyName, ICalParameters parameters, String value) throws IOException {
298                //validate the property name
299                if (invalidPropertyNameCharacters.containsAny(propertyName)) {
300                        throw Messages.INSTANCE.getIllegalArgumentException(8, propertyName, printableCharacterList(invalidPropertyNameCharacters.characters()));
301                }
302                if (beginsWithWhitespace(propertyName)) {
303                        throw Messages.INSTANCE.getIllegalArgumentException(9, propertyName);
304                }
305
306                value = sanitizeValue(parameters, value);
307
308                /*
309                 * Determine if the property value must be encoded in quoted printable
310                 * encoding. If so, then determine what charset to use for the encoding.
311                 */
312                boolean useQuotedPrintable = (parameters.getEncoding() == Encoding.QUOTED_PRINTABLE);
313                Charset quotedPrintableCharset = null;
314                if (useQuotedPrintable) {
315                        String charsetParam = parameters.getCharset();
316                        if (charsetParam == null) {
317                                quotedPrintableCharset = Charset.forName("UTF-8");
318                        } else {
319                                try {
320                                        quotedPrintableCharset = Charset.forName(charsetParam);
321                                } catch (Throwable t) {
322                                        quotedPrintableCharset = Charset.forName("UTF-8");
323                                }
324                        }
325                        parameters.setCharset(quotedPrintableCharset.name());
326                }
327
328                //write the property name
329                writer.append(propertyName);
330
331                //write the parameters
332                for (Map.Entry<String, List<String>> parameter : parameters) {
333                        String parameterName = parameter.getKey();
334                        List<String> parameterValues = parameter.getValue();
335                        if (parameterValues.isEmpty()) {
336                                continue;
337                        }
338
339                        if (version == ICalVersion.V1_0) {
340                                //e.g. ADR;FOO=bar;FOO=car:
341                                for (String parameterValue : parameterValues) {
342                                        parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
343                                        writer.append(';').append(parameterName).append('=').append(parameterValue);
344                                }
345                                continue;
346                        }
347
348                        //e.g. ADR;TYPE=home,work,"another,value":
349                        boolean first = true;
350                        writer.append(';').append(parameterName).append('=');
351                        for (String parameterValue : parameterValues) {
352                                if (!first) {
353                                        writer.append(',');
354                                }
355
356                                parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
357
358                                //surround with double quotes if contains special chars
359                                if (specialParameterCharacters.containsAny(parameterValue)) {
360                                        writer.append('"').append(parameterValue).append('"');
361                                } else {
362                                        writer.append(parameterValue);
363                                }
364
365                                first = false;
366                        }
367                }
368
369                writer.append(':');
370
371                //write the property value
372                writer.append(value, useQuotedPrintable, quotedPrintableCharset);
373                writer.append(writer.getNewline());
374        }

376        /**
377         * Determines if a given string starts with whitespace.
378         * @param string the string
379         * @return true if it starts with whitespace, false if not
380         */
381        private static boolean beginsWithWhitespace(String string) {
382                if (string.length() == 0) {
383                        return false;
384                }
385                char first = string.charAt(0);
386                return (first == ' ' || first == '\t');
387        }

389        /**
390         * Sanitizes a property value for safe inclusion in an iCalendar object.
391         * @param parameters the property's parameters
392         * @param value the value to sanitize
393         * @return the sanitized value
394         */
395        private String sanitizeValue(ICalParameters parameters, String value) {
396                if (value == null) {
397                        return "";
398                }
399
400                if (version == ICalVersion.V1_0 && newlineBitSet.containsAny(value)) {
401                        /*
402                         * 1.0 does not support the "\n" escape sequence (see "Delimiters"
403                         * sub-section in section 2 of the specs) so encode the value in
404                         * quoted-printable encoding if any newline characters exist.
405                         */
406                        parameters.setEncoding(Encoding.QUOTED_PRINTABLE);
407                        return value;
408                }
409
410                return escapeNewlines(value);
411        }
412
413        /**
414         * Sanitizes a parameter value for serialization.
415         * @param parameterValue the parameter value
416         * @param parameterName the parameter name
417         * @param propertyName the name of the property to which the parameter
418         * belongs
419         * @return the sanitized parameter value
420         * @throws IllegalArgumentException if the value contains invalid characters
421         */
422        private String sanitizeParameterValue(String parameterValue, String parameterName, String propertyName) {
423                CharacterBitSet invalidChars = (caretEncodingEnabled ? invalidParamValueCharsWithCaretEncoding : invalidParamValueChars).get(version);
424                if (invalidChars.containsAny(parameterValue)) {
425                        throw Messages.INSTANCE.getIllegalArgumentException(10, propertyName, parameterName, printableCharacterList(invalidChars.characters()));
426                }
427
428                String sanitizedValue = parameterValue;
429                switch (version) {
430                case V1_0:
431                        //Note: 1.0 does not support caret encoding.
432
433                        //escape backslashes
434                        sanitizedValue = sanitizedValue.replace("\\", "\\\\");
435
436                        //escape semi-colons (see section 2)
437                        return sanitizedValue.replace(";", "\\;");
438
439                default:
440                        if (caretEncodingEnabled) {
441                                return applyCaretEncoding(sanitizedValue);
442                        }
443                        return sanitizedValue;
444                }
445        }
446
447        private static String printableCharacterList(String list) {
448                return list.replace("\n", "\\n").replace("\r", "\\r");
449        }
450
451        /**
452         * Applies circumflex accent encoding to a string.
453         * @param value the string
454         * @return the encoded string
455         */
456        private String applyCaretEncoding(String value) {
457                value = value.replace("^", "^^");
458                value = newlineRegex.matcher(value).replaceAll("^n");
459                value = value.replace("\"", "^'");
460                return value;
461        }
462
463        /**
464         * Escapes all newlines in a string.
465         * @param string the string to escape
466         * @return the escaped string
467         */
468        private String escapeNewlines(String string) {
469                return newlineRegex.matcher(string).replaceAll("\\\\n");
470        }
